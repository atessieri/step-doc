  1"""CAM Message Version 1 Module.
  2
  3This module provides functionality for handling Cooperative Awareness Messages (CAM)
  4according to the ETSI ITS standard version 1. It includes the CamMessageV1 class
  5which manages the conversion, validation, and processing of CAM messages between
  6different formats including ASN.1, ORM (Object-Relational Mapping), dictionary,
  7and JSON representations.
  8
  9The module supports:
 10
 11    - Bi-directional conversion between ASN.1 and internal data formats
 12    - Access to key message components like position, station info, and vehicle parameters
 13    - Message validation and type safety through Pydantic models
 14    - Timestamp handling with ITS-specific time formats
 15    - Strict Python version requirements (3.11+)
 16
 17Dependencies:
 18
 19    - dpath: For accessing nested dictionary structures
 20    - step_cloud_schemas: For CAM message schemas and validation
 21    - typing: For type hints and overrides
 22"""

 49class CamMessageV1(FacilityMessage):
 50    """A class representing a CAM (Cooperative Awareness Message) message following
 51    version 1 format.
 52
 53    This class inherits from FacilityMessage and handles CAM messages which are used to
 54    share vehicle status information in V2X communications.
 55    """
 56
 57    __slots__: list[str] = ["_message"]
 58
 59    def __init__(
 60        self,
 61        message: Optional[CamDataV1] = None,
 62    ) -> None:
 63        """Initialize a CAM (Cooperative Awareness Message) V1 message container.
 64
 65        This constructor initializes a CAM message object with optional message
 66        content and timestamp.
 67
 68        Args:
 69            message (Optional[CamDataV1]): The CAM data content to encapsulate.
 70                Defaults to None.
 71
 72        Returns:
 73            None
 74        """
 75        super().__init__(
 76            its_time=(
 77                ItsTime.create_from_iso_time(time_str=message.creationTime)
 78                if message is not None
 79                else None
 80            )
 81        )
 82        self._message: Optional[CamDataV1] = message

 84    @override
 85    def __eq__(self, other: object) -> bool:
 86        """Checks if this CamMessageV1 instance is equal to another object.
 87
 88        The method first verifies if the other object is also a CamMessageV1 instance.
 89        Then it delegates to the parent class's equality check. If that passes,
 90        it compares the _message attributes of both objects.
 91
 92        Args:
 93            other (object): The object to compare with this instance.
 94
 95        Returns:
 96            bool: True if the objects are equal, False if they are not,
 97            or NotImplemented if other is not a CamMessageV1 instance.
 98        """
 99        if not isinstance(other, CamMessageV1):
100            return NotImplemented
101        super_eq = super().__eq__(other)
102        if super_eq is not True:
103            return super_eq
104        return self._message == other._message

195    def from_orm(self, data: CamDataV1) -> "CamMessageV1":
196        """Create a CamMessageV1 instance from an ORM data object.
197
198        This method initializes or updates the current instance with data from a CamDataV1
199        ORM object and optionally sets the message timestamp.
200
201        Args:
202            data (CamDataV1): The ORM data object containing CAM message information.
203
204        Returns:
205            CamMessageV1: The current instance, allowing for method chaining.
206
207        Raises:
