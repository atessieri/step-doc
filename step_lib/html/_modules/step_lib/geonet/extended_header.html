  1"""GeoNetworking Extended Header Implementation Module.
  2
  3This module implements the Extended Header component of the GeoNetworking protocol as
  4defined in ETSI TS 103 836-4-1 V2.1.1. The Extended Header follows the Common Header in the
  5GeoNetworking packet structure and contains specialized information depending on the
  6packet type.
  7
  8The module provides the following packet implementations:
  9
 10    - GeoUnicastPacket: For point-to-point delivery to a geographic position
 11    - SingleHopBroadcast: For delivery to all immediate neighbors (one hop)
 12    - GeoAnycastBroadcast: For delivery to nodes within a geographic area
 13
 14Each packet type supports serialization/deserialization to/from bytes and contains
 15fields as specified in the ETSI standard, such as:
 16
 17    - Position vectors for source and destination nodes
 18    - Sequence numbers for packet ordering
 19    - Geographic area definitions (for area-based delivery)
 20
 21The ExtendedHeader class acts as a container for the specific packet data and manages
 22sequence numbers for packet delivery.
 23"""

 69class GeoUnicastPacket(BasePacket):
 70    """Represents a GeoUnicast packet in the GeoNetworking protocol. GeoUnicast packets
 71    are used for point-to-point communication between two GeoNetworking nodes, where the
 72    destination is specified by its geographical position. The GeoUnicast packet

105    @staticmethod
106    def from_bytes_stream(stream: BytesIO) -> "GeoUnicastPacket":
107        """Deserializes a GeoUnicastPacket from a byte stream.
108
109        The method reads from the stream in the following order:
110
111            - Sequence Number (SN): 2 bytes, big-endian, unsigned
112            - Reserved: 2 bytes (skipped)
113            - Source Position Vector (SO PV): 24 bytes, long format
114            - Destination Position Vector (DE PV): 20 bytes, short format
115
116        Args:
117            stream (BytesIO): The byte stream to read from
118
119        Returns:
120            GeoUnicastPacket: A new instance containing the deserialized data
121
122        Note:
123            This method assumes that the stream pointer is positioned at the start
124            of the GeoUnicast header.
125        """
126        # SN (2 bytes)
127        sequence_received = int.from_bytes(
128            bytes=stream.read(2), byteorder="big", signed=False
129        )
130        # Reserved (2 bytes)
131        stream.read(2)
132        # SO PV (24 bytes)
133        long_position_vector_source = PositionVector.from_bytes_stream(
134            stream=stream, packet_type=PositionVectorType.LONG
135        )
136        # DE PV (20 bytes)
137        short_position_vector_destination = PositionVector.from_bytes_stream(
138            stream=stream, packet_type=PositionVectorType.SHORT
139        )
140        return GeoUnicastPacket(
141            sequence_received=sequence_received,
142            long_position_vector_source=long_position_vector_source,
143            short_position_vector_destination=short_position_vector_destination,
144        )

146    def to_bytes(self, sequence_sender: int) -> bytes:
147        """Converts the extended header to a bytes representation.
148
149        This method serializes the extended header fields into a binary format according to the
150        ETSI specification for GeoNetworking headers.
151
152        Args:
153            sequence_sender (int): The sequence number assigned by the sender.
154
155        Returns:
156            bytes: Binary representation of the extended header with the following structure:
157
158                - 2 bytes: Sequence Number (SN)
159                - 2 bytes: Reserved field (set to 0)
160                - 24 bytes: Source Position Vector (SO PV) in long format
161                - 20 bytes: Destination Position Vector (DE PV) in short format
162
163        Note:
164            The position vectors are serialized using their respective to_bytes methods.
165        """
166        stream = BytesIO()
167        # SN (2 bytes)
168        stream.write(int(sequence_sender).to_bytes(length=2, byteorder="big"))
169        # Reserved (2 bytes)
170        stream.write(b"\x00\x00")
171        # SO PV (24 bytes)
172        stream.write(
173            self.long_position_vector_source.to_bytes(
174                packet_type=PositionVectorType.LONG
175            )
176        )
177        # DE PV (20 bytes)
178        stream.write(
179            self.short_position_vector_destination.to_bytes(
180                packet_type=PositionVectorType.SHORT
181            )
182        )
183        return stream.getvalue()

312            - lat_destination is between -90.0 and 90.0
313            - lon_destination is between -180.0 and 180.0
314
315        Adjusts the following attributes to be within their min/max bounds:
316
317            - distance_a: clamped between 0 and 65535 in meters
318            - distance_b: clamped between 0 and 65535 in meters
319            - angle: clamped between 0 and 360 degrees
320
321        Raises:
322            ValueError: If latitude or longitude values are outside their valid ranges
323        """
324        if not self.LATITUDE_MIN <= self.lat_destination <= self.LATITUDE_MAX:
325            raise ValueError(
326                f"Destination latitude {self.lat_destination} must be between {self.LATITUDE_MIN} and {self.LATITUDE_MAX}."  # pylint: disable=line-too-long
327            )
328        if not self.LONGITUDE_MIN <= self.lon_destination <= self.LONGITUDE_MAX:

332        object.__setattr__(
333            self,
334            "distance_a",

348    @staticmethod
349    def from_bytes_stream(stream: BytesIO) -> "GeoAnycastBroadcast":
350        """Parse a GeoAnycastBroadcast object from a byte stream.
351
352        This method reads the following components from the stream:
353
354            - Sequence Number (SN): 2 bytes, unsigned, big-endian
355            - Reserved: 2 bytes (skipped)
356            - Source Position Vector (SO PV): 24 bytes, parsed as a long position vector
357            - GeoAreaPos Latitude: 4 bytes, signed, big-endian, scaled by LAT_LON_SCALE
358            - GeoAreaPos Longitude: 4 bytes, signed, big-endian, scaled by LAT_LON_SCALE
359            - Distance A: 2 bytes, unsigned, big-endian
360            - Distance B: 2 bytes, unsigned, big-endian
361            - Angle: 2 bytes, unsigned, big-endian
362            - Reserved: 2 bytes (skipped)
363
364        Args:
365            stream (BytesIO): The byte stream to read from.
366
367        Returns:
368            GeoAnycastBroadcast: An instance of GeoAnycastBroadcast with the parsed values.
369        """
370        # SN (2 bytes)
371        sequence_received = int.from_bytes(
372            bytes=stream.read(2), byteorder="big", signed=False
373        )
374        # Reserved (2 bytes)
375        stream.read(2)
376        # SO PV (24 bytes)
377        long_position_vector_source = PositionVector.from_bytes_stream(
378            stream=stream, packet_type=PositionVectorType.LONG
379        )
380        # GeoAreaPos Latitude (4 bytes)
381        latitude = (
382            int.from_bytes(bytes=stream.read(4), byteorder="big", signed=True)
383            / GeoAnycastBroadcast.LAT_LON_SCALE
384        )
385        # GeoAreaPos Longitude (4 bytes)
386        longitude = (
387            int.from_bytes(bytes=stream.read(4), byteorder="big", signed=True)
388            / GeoAnycastBroadcast.LAT_LON_SCALE
389        )
390        # Distance A (2 bytes)
391        distance_a = int.from_bytes(bytes=stream.read(2), byteorder="big", signed=False)
392        # Distance B (2 bytes)
393        distance_b = int.from_bytes(bytes=stream.read(2), byteorder="big", signed=False)
394        # Angle (2 bytes)
395        angle = int.from_bytes(bytes=stream.read(2), byteorder="big", signed=False)
396        # Reserved (2 bytes)
397        stream.read(2)
398        return GeoAnycastBroadcast(
399            sequence_received=sequence_received,
400            long_position_vector_source=long_position_vector_source,
401            lat_destination=latitude,
402            lon_destination=longitude,
403            distance_a=distance_a,
404            distance_b=distance_b,
405            angle=angle,
406        )

408    def to_bytes(self, sequence_sender: int) -> bytes:
409        """Convert the GeonetExtendedHeader to bytes for network transmission.
410
411        The method serializes the extended header according to the
412        ETSI TS 103 836-4-1 V2.1.1 specification:
413
414            - Sequence Number (SN): 2 bytes, big-endian
415            - Reserved field: 2 bytes, set to 0
416            - Source Position Vector (SO PV): 24 bytes, long format
417            - Destination Latitude: 4 bytes, big-endian, signed, scaled by LAT_LON_SCALE
418            - Destination Longitude: 4 bytes, big-endian, signed, scaled by LAT_LON_SCALE
419            - Distance A: 2 bytes, big-endian
420            - Distance B: 2 bytes, big-endian
421            - Angle: 2 bytes, big-endian
422            - Reserved field: 2 bytes, set to 0
423
424        Args:
425            sequence_sender (int): The sequence number of the sender
426
427        Returns:
428            bytes: The serialized extended header as bytes
429        """
430        stream = BytesIO()
431        # SN (2 bytes)
432        stream.write(int(sequence_sender).to_bytes(length=2, byteorder="big"))
433        # Reserved (2 bytes)
434        stream.write(b"\x00\x00")
435        # SO PV (24 bytes)
436        stream.write(
437            self.long_position_vector_source.to_bytes(
438                packet_type=PositionVectorType.LONG
439            )
440        )
441        # GeoAreaPos Latitude (4 bytes)
442        stream.write(
443            int(self.lat_destination * self.LAT_LON_SCALE).to_bytes(
444                length=4, byteorder="big", signed=True
445            )
446        )
447        # GeoAreaPos Longitude (4 bytes)
448        stream.write(
449            int(self.lon_destination * self.LAT_LON_SCALE).to_bytes(
450                length=4, byteorder="big", signed=True
451            )
452        )
453        # Distance A (2 bytes)
454        stream.write(int(self.distance_a).to_bytes(length=2, byteorder="big"))
455        # Distance B (2 bytes)
456        stream.write(int(self.distance_b).to_bytes(length=2, byteorder="big"))
457        # Angle (2 bytes)
458        stream.write(int(self.angle).to_bytes(length=2, byteorder="big"))
459        # Reserved (2 bytes)
460        stream.write(b"\x00\x00")
461        return stream.getvalue()

478    The Extended Header contains specific packet data based on the type of GeoNetworking packet

484            The specific packet data contained in this extended header.
485    """

498        Reads the appropriate packet data based on the provided packet type and

501        Args:

506        Returns:

513        match packet_type:
514            case HeaderTypeSubtype.GEOUNICAST:

522                | HeaderTypeSubtype.GEOBROADCAST_CIRCLE

525            ):
526                packet_data = GeoAnycastBroadcast.from_bytes_stream(stream=stream)
527            case _:
528                raise ValueError(f"Invalid packet type: {packet_type}")
529        return ExtendedHeader(packet_data=packet_data)

531    def to_bytes(self, packet_type: HeaderTypeSubtype) -> bytes:
532        """Convert the extended header to a byte representation.
533
534        This method serializes the extended header based on the packet type. It validates
535        that the packet_data attribute is of the correct type for the given packet_type,
536        and then delegates the serialization to the appropriate packet class.
537
538        For GEOUNICAST and GEO(ANY/BROAD)CAST types, the sequence_sender is incremented
539        after serialization.
540
541        Args:
542            packet_type: The type and subtype of the packet to serialize
543
544        Returns:
545            The byte representation of the extended header
546
547        Raises:
548            ValueError: If the packet_data attribute is not of the expected type for the
549                given packet_type, or if the packet_type is not supported
550        """
551        match packet_type:
552            case HeaderTypeSubtype.GEOUNICAST:
553                if not isinstance(self.packet_data, GeoUnicastPacket):
554                    raise ValueError(
555                        f"Invalid packet data type for {packet_type}: {type(self.packet_data)}"  # pylint: disable=line-too-long
556                    )
557                data = self.packet_data.to_bytes(sequence_sender=self.sequence_sender)
558                ExtendedHeader.sequence_sender = (
559                    ExtendedHeader.sequence_sender + 1
560                ) % ExtendedHeader.SEQUENCE_MODULE
561            case HeaderTypeSubtype.SINGLE_HOP:
562                if not isinstance(self.packet_data, SingleHopBroadcast):
563                    raise ValueError(
564                        f"Invalid packet data type for {packet_type}: {type(self.packet_data)}"  # pylint: disable=line-too-long
565                    )
566                data = self.packet_data.to_bytes()
567            case (
568                HeaderTypeSubtype.GEOANYCAST_CIRCLE
569                | HeaderTypeSubtype.GEOANYCAST_RECTANGLE
570                | HeaderTypeSubtype.GEOANYCAST_ELLIPSE
571                | HeaderTypeSubtype.GEOBROADCAST_CIRCLE
572                | HeaderTypeSubtype.GEOBROADCAST_RECTANGLE
573                | HeaderTypeSubtype.GEOBROADCAST_ELLIPSE
574            ):
575                if not isinstance(self.packet_data, GeoAnycastBroadcast):
576                    raise ValueError(
577                        f"Invalid packet data type for {packet_type}: {type(self.packet_data)}"  # pylint: disable=line-too-long
578                    )
579                data = self.packet_data.to_bytes(sequence_sender=self.sequence_sender)
580                ExtendedHeader.sequence_sender = (
581                    ExtendedHeader.sequence_sender + 1
582                ) % ExtendedHeader.SEQUENCE_MODULE
583            case _:
584                raise ValueError(f"Invalid packet data type: {packet_type}")
585        return data
