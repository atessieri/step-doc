  1"""Geo-Networking Address Module for V2X Communications.
  2
  3This module provides the implementation of the GeoNetworkAddress class according to the
  4ETSI TS 103 836-4-1 specification for Intelligent Transport Systems (ITS).
  5
  6The GeoNetworkAddress is a fundamental component of the GeoNetworking protocol stack
  7used in Vehicle-to-Everything (V2X) communications. It uniquely identifies network nodes
  8in a geographical area, supporting location-aware routing and addressing.
  9
 10Key features:
 11
 12    - Implements the 8-byte GeoNetworkAddress format as specified by ETSI
 13    - Supports serialization to and deserialization from binary format
 14    - Validates address components according to protocol specifications
 15    - Integrates with the TrafficParticipantType enumeration for node classification
 16
 17The address structure contains:
 18
 19    - Manual configuration flag (1 bit)
 20    - Traffic Participant Type (5 bits)
 21    - Reserved bits (10 bits)
 22    - MAC ID (48 bits/6 bytes)
 23
 24This module is designed to be used as part of the ITS GeoNetworking stack
 25in V2X communication systems.
 26"""

 38@dataclass(frozen=True, kw_only=True, slots=True)
 39class GeoNetworkAddress:
 40    """
 41    GeoNetworkAddress class representing a Geo-Networking address according to ETSI
 42    TS 103 836-4-1 specification.
 43    This class implements the GeoNetworking address format which includes:
 44
 45        - A manual configuration flag (M)
 46        - Traffic participant type (TPT)
 47        - MAC ID (48-bit identifier)
 48
 49    The GeoNetworkAddress is an 8-byte structure where:
 50
 51        - First 2 bytes contain the M flag (1 bit), TPT (5 bits), and reserved bits (10 bits)
 52        - Last 6 bytes contain the MAC ID in network byte order (big endian)
 53
 54    Attributes:
 55        is_manual_configured (bool): Flag indicating if the address is manually configured
 56        traffic_participant_type (TrafficParticipantType): Type of traffic participant
 57        mac_id (int): 48-bit MAC identifier
 58
 59    Raises:
 60        ValueError: If MAC ID is outside the valid range (0 to MAX_MAC_ID)
 61    """
 62
 63    MAC_ID_MIN: ClassVar[int] = 0
 64    MAC_ID_MAX: ClassVar[int] = 0xFFFFFFFFFFFF
 65
 66    is_manual_configured: bool = True
 67    traffic_participant_type: TrafficParticipantType
 68    mac_id: int

 80    @staticmethod
 81    def from_bytes_stream(stream: BytesIO) -> "GeoNetworkAddress":
 82        """Deserializes a GeoNetworkAddress from a bytes stream.
 83
 84        The byte structure is as follows:
 85
 86            - M (1 bit): Manual configuration flag
 87            - TPT (5 bits): Traffic Participant Type
 88            - Reserved (10 bits): Set to zero
 89            - MAC ID (6 bytes): MAC address in network byte order (big endian)
 90
 91        Args:
 92            stream (BytesIO): A bytes input stream positioned at the start of a GeoNetworkAddress.
 93
 94        Returns:
 95            GeoNetworkAddress: A new GeoNetworkAddress instance constructed from the stream data.
 96
 97        Note:
 98            The stream pointer is advanced by 8 bytes after this operation.
 99        """
100        bits = ConstBitStream(bytes=stream.read(2))
101        # M (1 bit)
102        is_manual_configured = bool(bits.read(fmt="bool"))
103        # TPT (5 bits)
104        traffic_participant_type = TrafficParticipantType.from_int(
105            value=bits.read(fmt="uint:5")
106        )
107        # Reserved (10 bits)
108        # MAC ID
109        mac_id = int.from_bytes(stream.read(6), "big")
110        return GeoNetworkAddress(
111            is_manual_configured=is_manual_configured,
112            traffic_participant_type=traffic_participant_type,
113            mac_id=mac_id,
114        )

116    def to_bytes(self) -> bytes:
117        """Convert GeoNetworkAddress to bytes.
118
119        This method serializes the GeoNetworkAddress object to a byte string according to the ETSI
120        TS 103 836-4-1 GeoNetworking protocol specification.
121
122        The byte structure is as follows:
123
124            - M (1 bit): Manual configuration flag
125            - TPT (5 bits): Traffic Participant Type
126            - Reserved (10 bits): Set to zero
127            - MAC ID (6 bytes): MAC address in network byte order (big endian)
128
129        Returns:
130            bytes: The serialized GeoNetworkAddress as a byte string (8 bytes total)
131        """
132        # M (1 bit) + TPT (5 bits) + Reserved (10 bits)
133        bits = BitStream()
134        # M (1 bit)
135        bits.append(f"bool:1={self.is_manual_configured}")
136        # TPT (5 bits)
137        traffic_participant_code = TrafficParticipantType.to_int(
138            participant_type=self.traffic_participant_type
139        )
140        bits.append(f"uint:5={traffic_participant_code}")
141        # Reserved (10 bits)
142        bits.append("uint:10=0")
143        # MAC ID
144        return bits.tobytes() + self.mac_id.to_bytes(length=6, byteorder="big")
