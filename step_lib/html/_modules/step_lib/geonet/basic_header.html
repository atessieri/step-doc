  1"""GeoNetworking Basic Header Implementation Module.
  2
  3This module provides classes and utilities for working with the GeoNetworking Basic Header
  4as specified in ETSI TS 103 836-4-1 V2.1.1. The Basic Header is the first part of every
  5GeoNetworking packet and contains essential routing and protocol information.
  6
  7The module includes:
  8
  9    - Enumerations for protocol versions and header types (GeonetVersion, NextHeader)
 10    - Time management utilities for packet lifetime (LifeTimeBase, LifeTime)
 11    - Implementation of the BasicHeader class with serialization/deserialization functionality
 12
 13The Basic Header structure contains:
 14
 15    - Protocol version (4 bits)
 16    - Next Header type indicator (4 bits)
 17    - Reserved field (8 bits)
 18    - Lifetime value (8 bits)
 19    - Remaining hop limit (8 bits)
 20
 21This module supports parsing from byte streams and conversion to byte representation
 22for network transmission in GeoNetworking applications.
 23"""

106@dataclass(frozen=True, kw_only=True, slots=True)
107class LifeTime:
108    """
109    A class for managing and representing lifetime values in GEONET protocol.
110    The LifeTime class is responsible for handling lifetime values, which are durations
111    expressed in a compact form using a combination of a multiplier and a base value.
112    This encoding allows for representing a wide range of time intervals using a single byte.
113    This class provides functionality to:
114
115        - Create lifetime values from byte streams
116        - Convert lifetime values to their byte representation
117        - Calculate appropriate base values and multipliers for efficient encoding
118
119    Attributes:
120        value (int): The actual lifetime value, with a default of 950.
121    """
122
123    MAX_MULTIPLIER: ClassVar[int] = 63
124    _BASE_VALUES: ClassVar[Dict[LifeTimeBase, int]] = {
125        base: LifeTimeBase.base_to_value(base=base) for base in LifeTimeBase
126    }
127    _MAX_VALUES: ClassVar[Dict[LifeTimeBase, int]] = {
128        base: LifeTimeBase.base_to_value(base=base) * 63 for base in LifeTimeBase
129    }
130
131    value: int = 950

133    @staticmethod
134    def from_byte_stream(stream: BytesIO) -> "LifeTime":
135        """Create a LifeTime object from a byte stream.
136
137        This method reads one byte from the provided byte stream and interprets it as follows
138        starting from the most significant bit (MSB) big endian:
139
140            - The first 6 bits represent the multiple value
141            - The last 2 bits represent the LifeTimeBase
142
143        The final LifeTime value is calculated by multiplying the multiple value with
144        the corresponding base value derived from the LifeTimeBase.
145
146        Args:
147            stream (BytesIO): The byte stream to read from
148
149        Returns:
150            LifeTime: A new LifeTime object with the calculated value
151
152        Note:
153            This method advances the stream's position by one byte.
154        """
155        byte_val = stream.read(1)[0]
156        # Extract multiple (6 most significant bits) and base (2 least significant bits)
157        multiple = byte_val >> 2
158        base = LifeTimeBase(byte_val & 0x03)
159        return LifeTime(value=multiple * LifeTime._BASE_VALUES[base])

161    @lru_cache(maxsize=256)
162    def to_bytes(self) -> bytes:
163        """Converts the lifetime value to a byte representation.
164
165        The method encodes the lifetime value using a combination of a multiplier and a base value.
166        It first determines the appropriate base by iterating through available lifetime base values
167        until finding one that, when multiplied by the maximum allowed multiplier, is greater than
168        or equal to the current lifetime value.
169
170        Then it calculates the actual multiplier for the chosen base (capped to MAX_MULTIPLIER)
171        and creates a bit-stream with 6 bits (MSB big endian) for the multiplier and 2 bits
172        (big endian) for the base identifier.
173
174        Returns:
175            bytes: A bytes object containing the encoded lifetime value (1 byte total).
176        """
177        # Find appropriate base using pre-calculated max values
178        for base, max_value in self._MAX_VALUES.items():
179            if max_value >= self.value:
180                current_base = base
181                break
182        else:
183            # If no suitable base found, use the largest available
184            current_base = LifeTimeBase.X100S
185
186        base_value = self._BASE_VALUES[current_base]
187        multiplier = min(self.value // base_value, self.MAX_MULTIPLIER)
188
189        # Combine multiplier (6 bits) and base (2 bits) directly
190        byte_val = (multiplier << 2) | current_base.value
191        return bytes([byte_val])

216    def __post_init__(self) -> None:
217        """Validates the `remaining_hop_limit` attribute after instance initialization.
218
219        This method is automatically called after init.
220        It ensures that the remaining hop limit is within the allowed range defined by
221        the class constants REMAINING_HOP_LIMIT_MIN and REMAINING_HOP_LIMIT_MAX.
222
223        Raises:
224            ValueError: If remaining_hop_limit is outside the valid range.
225        """

236    def from_byte_stream(stream: BytesIO) -> "BasicHeader":
237        """Parse a BasicHeader from a byte stream.
238
239        This method reads from the provided byte stream to construct a BasicHeader object.
240        The byte stream is expected to contain:
241
242            - 1 byte in msb:
243
244                - 4 bits for version big endian
245                - 4 bits for next_header big endian
246
247            - 1 byte reserved (skipped)
248            - LifeTime object bytes (read via LifeTime.from_byte_stream)
249            - 1 byte for remaining_hop_limit
250
251        Args:
252            stream (BytesIO): The byte stream to read from.
253
254        Returns:
255            BasicHeader: A new BasicHeader object constructed from the byte stream data.
256
257        Raises:
258            ValueError: If the next_header field indicates a SECURED_PACKET, which is not supported.
259        """
260        value = stream.read(1)[0]
261        # Version
262        version = GeonetVersion(value >> 4)
263        # NH
264        next_header = NextHeader(value & 0x0F)
265        if next_header in (NextHeader.SECURED_PACKET, NextHeader.ANY):
266            raise ValueError(
267                f"Basic header next header: {next_header} is not supported"
268            )
269        # Reserved
270        stream.read(1)
271        # LT
272        life_time = LifeTime.from_byte_stream(stream=stream)
273        # RHL
274        remaining_hop_limit = stream.read(1)[0]
275        return BasicHeader(
276            version=version,
277            next_header=next_header,
278            life_time=life_time,
279            remaining_hop_limit=remaining_hop_limit,
280        )

282    def to_bytes(self) -> bytes:
283        """Convert the basic header to bytes.
284
285        This method serializes the basic header into a byte stream according to the GeoNet protocol.
286        The serialization includes most significant bit (MSB) ordering for the first byte,
287        which contains:
288
289            - Version (4 bits)
290            - Next Header type (NH) (4 bits)
291            - Reserved field (8 bits, set to 0x00)
292            - Lifetime (LT) (8 bits)
293            - Remaining hop limit (RHL) (8 bits)
294
295        Returns:
296            bytes: The serialized basic header as bytes
297        """
298        # Combine version and next_header directly
299        header_byte = (self.version.value << 4) | self.next_header.value
300        # Concatenate all bytes directly without using BytesIO for better performance
301        return (
302            bytes(
303                [
304                    header_byte,  # Version and NH
305                    0,  # Reserved
306                ]
307            )
308            + self.life_time.to_bytes()
309            + bytes([self.remaining_hop_limit])
310        )
