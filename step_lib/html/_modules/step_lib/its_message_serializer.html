  1"""ITS Message Serialization Module.
  2
  3This module provides functionality for serializing and deserializing Intelligent
  4Transport Systems (ITS) messages following the ETSI standards. It implements the
  5complete protocol stack consisting of:
  6
  7    - GeoNetworking layer - Geographic addressing and routing
  8    - BTP (Basic Transport Protocol) layer - Transport layer services
  9    - Facility layer - Application-specific message formats (CAM, DENM, CPM, VAM)
 10
 11The module offers:
 12
 13    - Conversion between different message formats and protocols
 14    - Serialization of ITS messages for transmission over the network
 15    - Deserialization of received ITS messages
 16    - Support for various geographic addressing modes (single-hop, broadcast, etc.)
 17    - GPS position representation and handling
 18
 19This is a core component of the STEP library, supporting V2X (Vehicle-to-Everything)
 20communications in intelligent transportation systems.
 21"""

 90class ItsMessageSerializer:
 91    """ITS Message Serializer for GeoNet, BTP, and Facility communications. This class
 92    handles serialization and deserialization of ITS (Intelligent Transport Systems)
 93    messages, which consist of three layers:
 94
 95        - GeoNet (Geographic Networking)
 96        - BTP (Basic Transport Protocol)
 97        - Facility (Application) layer
 98
 99    It provides mapping between message types, ports, and headers, and handles the creation
100    and parsing of properly formatted ITS protocol messages.
101    The class maintains cache mechanisms to optimize performance when creating messages
102    with similar configurations.
103
104    Attributes:
105        logger: Logger instance for logging operations
106        instance_name: Name of the instance
107        station_id: Optional station identifier
108        cache_dir: Optional directory path for caching
109
110    Note:

112        and VAM message types.

173    def _facility_message_id_to_btp_port(
174        self, message_type: MessageId
175    ) -> BtpWellKnownPort:
176        """Maps a message type to its corresponding BTP port number.
177
178        This method uses a predefined mapping dictionary to convert message IDs
179        to their associated port numbers for BTP (Binary Transport Protocol) communication.
180
181        Args:
182            message_type (MessageId): The message type identifier to convert to a port number.
183
184        Returns:

186
187        Raises:
188            ValueError: If the message type is not supported (not found in the mapping
189                dictionary).
190        """
191        try:
192            return self._MESSAGE_ID_TO_PORT_MAP[message_type]
193        except KeyError as exc:
194            raise ValueError(f"Unsupported message type: {message_type}") from exc

256    def deserialize(
257        self, message: bytes
258    ) -> tuple[GeonetMessage, BtpMessage, dict[Any, Any]]:
259        """Deserialize a ITS message into its component parts.
260
261        This method takes a byte array representing an ITS message and deserializes it
262        into its component parts: a GeoNet message, a BTP message, and a facility message.
263
264        Args:
265            message (bytes): The raw ITS message bytes to deserialize.
266
267        Returns:
268            tuple[GeonetMessage, BtpMessage, dict[Any, Any]]: A tuple containing:
269
270                - The deserialized GeoNet message
271                - The deserialized BTP message
272                - The deserialized facility message as a dictionary
273
274        Raises:
275            Any exceptions raised by the underlying serializers when parsing invalid data.
276        """
277        self._logger.debug(msg=f"Deserializing message: {message.hex()}")
278        stream = BytesIO(initial_bytes=message)
279        geonet_message = GeonetMessage.from_byte_stream(stream=stream)
280        btp_message = BtpMessage.from_bytes_stream(
281            byte_stream=stream,
282            btp_type=self._common_next_header_to_btp_type(
283                next_header=geonet_message.common_header.next_header
284            ),
285        )
286        facility_message = self._its_facility_serializer.deserialize(
287            message=stream.read(None),
288            message_type=self.btp_port_to_facility_message_id(
289                port=BtpWellKnownPort(btp_message.destination_port)
290            ),
291        )
292        self._logger.debug(msg=f"Serializing geonet message: {geonet_message}")
293        self._logger.debug(msg=f"Serializing btp message: {btp_message}")
294        self._logger.debug(msg=f"Serializing facility message: {facility_message}")
295        return (geonet_message, btp_message, facility_message)

327        Args:
328            address (GeoNetworkAddress): The GeoNetwork address to be included in the position

375    def _create_common_extended_anycast_broadcast(
376        self,
377        facility_message: FacilityMessage,
378        address: GeoNetworkAddress,
379        current_position: GpsLocation,
380        next_header: common_NextHeader,
381        payload_length: int,
382        header_type_subtype: HeaderTypeSubtype,
383        is_accurate: bool,
384        is_station_mobile: bool,
385    ) -> tuple[CommonHeader, ExtendedHeader]:
386        """Create common and extended headers for GeoAnycast and GeoBroadcast messages.
387
388        This method generates the headers required for geographic addressing messages,
389        specifically for circle, rectangle, and ellipse shapes for both anycast and broadcast.
390
391        Note: Despite accepting different header types, this method currently forces the
392        header type to GEOBROADCAST_CIRCLE.
393
394        Args:
395            facility_message: Contains the reference position for the destination
396            address: GeoNetwork address used for the source position vector
397            current_position: Current GPS location of the source
398            next_header: The next header type according to the GeoNetworking protocol
399            payload_length: Length of the payload after the GeoNetworking header
400            header_type_subtype: Type and subtype of the GeoNetworking header (must be a
401                GeoAnycast or GeoBroadcast type)
402            is_accurate: Indicates if the position of the source is accurate
403            is_station_mobile: Indicates if the station is mobile
404
405        Returns:
406            tuple[CommonHeader, ExtendedHeader]: A tuple containing:
407
408                - CommonHeader: The common header for the GeoNetworking message
409                - ExtendedHeader: The extended header specific to GeoAnycast/GeoBroadcast
410                  messages
411
412        Raises:
413            ValueError: If an unsupported header_type_subtype is provided
414        """
415        if header_type_subtype not in (
416            HeaderTypeSubtype.GEOANYCAST_CIRCLE,
417            HeaderTypeSubtype.GEOANYCAST_RECTANGLE,
418            HeaderTypeSubtype.GEOANYCAST_ELLIPSE,
419            HeaderTypeSubtype.GEOBROADCAST_CIRCLE,
420            HeaderTypeSubtype.GEOBROADCAST_RECTANGLE,
421            HeaderTypeSubtype.GEOBROADCAST_ELLIPSE,
422        ):
423            raise ValueError(f"Wrong header type subtype: {header_type_subtype}")
424        return (
425            CommonHeader(
426                next_header=next_header,
427                header_type_subtype=header_type_subtype,
428                traffic_class=TrafficClass(
429                    must_buffered=False,
430                    identification=TrafficClassIdentifierKnown.TC_LTE_V2X_HIGH_PRIORITY_DENM.value,  # pylint: disable=line-too-long
431                ),
432                flags=common_Flags(is_station_mobile=is_station_mobile),
433                payload_length_after_geonet=payload_length,
434                maximum_hop_limit=1,
435            ),
436            ExtendedHeader(
437                packet_data=GeoAnycastBroadcast(
438                    long_position_vector_source=PositionVector(
439                        address=address,
440                        time=current_position.time,
441                        latitude=current_position.latitude,
442                        longitude=current_position.longitude,
443                        is_accurate=is_accurate,
444                        speed=current_position.speed,
445                        heading=current_position.heading,
446                    ),
447                    lat_destination=facility_message.reference_position.latitude,
448                    lon_destination=facility_message.reference_position.longitude,
449                    distance_a=200,
450                    distance_b=(
451                        200
452                        if header_type_subtype
453                        not in (
454                            HeaderTypeSubtype.GEOBROADCAST_CIRCLE,
455                            HeaderTypeSubtype.GEOANYCAST_CIRCLE,
456                        )
457                        else 0
458                    ),
459                    angle=0,
460                )
461            ),
462        )

464    def _create_geonet_message(
465        self,
466        facility_message: FacilityMessage,
467        address: GeoNetworkAddress,
468        current_position: GpsLocation,
469        next_header: common_NextHeader,
470        payload_length: int,
471        is_station_mobile: bool,
472    ) -> GeonetMessage:
473        """Creates a GeonetMessage based on the given facility message.
474
475        This method constructs a GeonetMessage with appropriate headers depending
476        on the message type.
477
478        Args:
479            facility_message: The facility message containing the message type.
480            address: The GeoNetwork address information.
481            current_position: The current GPS location.
482            next_header: The next header type for the common header.
483            payload_length: Length of the payload in bytes.
484            is_station_mobile: Boolean indicating if the station is mobile.
485
486        Returns:
487            GeonetMessage: A fully constructed GeonetMessage with basic, common,
488                and extended headers.
489
490        Raises:
491            StepLibException: If the message type is not supported.
492
493        Note:
494            Position accuracy is determined by comparing the GPS accuracy with
495            ACCURATE_POSITION_MAX_ERROR.
496            Different header combinations are created based on the message type:
497
498                - CAM_PDU and CPM_PDU use single hop headers
499                - DENM_PDU uses anycast broadcast headers
500        """
501        is_accurate: bool
502        if current_position.accuracy is None:
503            is_accurate = False
504        elif current_position.accuracy <= self.ACCURATE_POSITION_MAX_ERROR:
505            is_accurate = True
506        else:
507            is_accurate = False
508        match facility_message.message_type:
509            case MessageId.CAM_PDU | MessageId.CPM_PDU | MessageId.CPM_PDU:
510                (common_header, extended_header) = (
511                    self._create_common_extended_single_hop(
512                        address=address,
513                        current_position=current_position,
514                        next_header=next_header,
515                        payload_length=payload_length,
516                        is_accurate=is_accurate,
517                        is_station_mobile=is_station_mobile,
518                    )
519                )
520            case MessageId.DENM_PDU:
521                (common_header, extended_header) = (
522                    self._create_common_extended_anycast_broadcast(
523                        facility_message=facility_message,
524                        address=address,
525                        current_position=current_position,
526                        next_header=next_header,
527                        payload_length=payload_length,
528                        header_type_subtype=HeaderTypeSubtype.GEOBROADCAST_CIRCLE,
529                        is_accurate=is_accurate,
530                        is_station_mobile=is_station_mobile,
531                    )
532                )
533            case _:
534                raise StepLibException(
535                    f"Unsupported message type: {facility_message.message_type}"
536                )
537        return GeonetMessage(
538            basic_header=self._BASIC_HEADER,
539            common_header=common_header,
540            extended_header=extended_header,
541        )

543    def serialize(
544        self,
545        facility_message: FacilityMessage,
546        address: GeoNetworkAddress,
547        current_position: GpsLocation,
548        is_station_mobile: bool,
549    ) -> bytes:
550        """Serialize a facility message into bytes for transmission.
551
552        This method serializes a facility message along with address, position,
553        and mobility information into a complete ITS message byte structure.
554        The serialization follows the ITS protocol stack
555        by combining facility, BTP, and GeoNetworking layers.
556
557        Args:
558            facility_message (FacilityMessage): The facility layer message to be serialized.
559            address (GeoNetworkAddress): The GeoNetworking address information.
560            current_position (GpsLocation): The current GPS position of the station.
561            is_station_mobile (bool): Flag indicating whether the station is mobile (True)
562                or stationary (False).
563
564        Returns:
565            bytes: The complete serialized message as a byte array containing the GeoNetwork header,
566            BTP header and facility message payload.
567
568        Logging:
569            Debug logs are generated for input parameters and the final serialized message.
570        """
571        self._logger.debug(msg=f"Serializing facility: {facility_message}")
572        self._logger.debug(msg=f"Address: {address}")
573        self._logger.debug(msg=f"Current position: {current_position}")
574        self._logger.debug(msg=f"Is station mobile: {is_station_mobile}")
575        facility_message_bytes = self._its_facility_serializer.serialize(
576            message=facility_message
577        )
578        btp_message = BtpMessage(
579            destination_port=self._facility_message_id_to_btp_port(
580                message_type=facility_message.message_type
581            ),
582            destination_port_info=0,
583        )
584        btp_message_bytes = btp_message.to_bytes()
585        geonet_message_bytes = self._create_geonet_message(
586            facility_message=facility_message,
587            address=address,
588            current_position=current_position,
589            next_header=self._btp_type_to_common_next_header(
590                btp_type=btp_message.get_btp_type()
591            ),
592            payload_length=len(btp_message_bytes) + len(facility_message_bytes),
593            is_station_mobile=is_station_mobile,
594        ).to_bytes()
595        # Combine all bytes at once instead of multiple joins
596        message = bytearray(geonet_message_bytes)
597        message.extend(btp_message_bytes)
598        message.extend(facility_message_bytes)
599        self._logger.debug(msg=f"Serialized message: {bytes(message).hex()}")
600        return bytes(message)
