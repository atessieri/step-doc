  1"""This module provides asynchronous MQTT communication functionality for ITS
  2(Intelligent Transport Systems) messages.
  3
  4The module implements a custom MQTT client that handles the publishing and receiving
  5of ITS messages through a broker using the MQTT v5 protocol.
  6It includes:
  7
  8    - MqttQoS: Enumeration for MQTT Quality of Service levels
  9    - PublishTopicParams: Configuration dataclass for MQTT topic publishing parameters
 10    - AsyncComManagerParams: Configuration dataclass for the MQTT client settings
 11    - AsyncComManager: Main class extending aiomqtt.Client with ITS message
 12      handling capabilities
 13
 14The implementation supports:
 15
 16    - Asynchronous message publishing and subscription
 17    - Geohash-based topic organization
 18    - Message serialization and deserialization
 19    - Configurable QoS levels
 20    - TLS/SSL security
 21    - Comprehensive logging
 22    - Error handling for message processing
 23
 24Example:
 25    params = AsyncComManagerParams.get_default()
 26    client = AsyncComManager(params)
 27    await client.connect()
 28    await client.publish_its_message(message=message, is_station_mobile=True)
 29"""

 66class MqttQoS(IntEnum):
 67    """
 68    MqttQoS: Enumeration class for MQTT Quality of Service levels.
 69
 70    The MQTT protocol provides three QoS levels for message delivery:
 71
 72    Attributes:
 73        QOS_0: At most once delivery. The message is delivered according to the best efforts
 74            of the operating environment. Message loss can occur.
 75        QOS_1: At least once delivery. The message is assured to arrive but duplicates
 76            may occur.
 77        QOS_2: Exactly once delivery. The message is assured to arrive exactly once. This is
 78            the highest quality of service level and has the highest overhead.
 79    """
 80
 81    QOS_0 = 0
 82    QOS_1 = 1
 83    QOS_2 = 2

131class AsyncComManagerParams:  # pylint: disable=too-many-instance-attributes
132    """This class manages configuration parameters for asynchronous MQTT communication.
133    This class encapsulates various configuration parameters needed for setting up and
134    managing asynchronous MQTT connections. It includes settings for connection details,
135    protocol behavior, message handling, security, and performance optimization.
136
137    Attributes:
138        instance_name (str): Unique identifier for the MQTT client instance.
139        host (str): MQTT broker hostname or IP address.
140        port (int): MQTT broker port number.
141        username (str): Authentication username for the MQTT broker.
142        password (str): Authentication password for the MQTT broker.
143        station_id (Optional[int]): Station ID for the client. This parameter is used to filter
144            messages received if the id is equal and override the station id in the published
145            messages. If None it isn't used. Defaults to None.
146        timeout (int): Connection timeout in seconds. Defaults to 30.
147        keepalive (int): Keep-alive interval in seconds. Defaults to 60.
148        bind_address (str): Local address to bind to. Defaults to empty string.
149        bind_port (int): Local port to bind to. Defaults to 0.
150        clean_start (CleanStartOption): Clean session flag. Defaults to True.
151        no_local (bool): If the value is 1, Application Messages must not be forwarded to a
152            connection with a ClientID equal to the ClientID of the publishing connection.
153            Defaults to True.
154        queue_type (Optional[type[AsyncioQueue[AiomqttMessage]]]): Queue type for message handling.
155        max_queued_incoming_messages (int): Maximum number of queued incoming messages.
156            Defaults to 60.
157        max_queued_outgoing_messages (int): Maximum number of queued outgoing messages.
158            Defaults to 60.
159        max_inflight_messages (int): Maximum number of in-flight messages. Defaults to 60.
160        max_concurrent_outgoing_calls (int): Maximum number of concurrent outgoing calls.
161            Defaults to 60.
162        tls_context (Optional[SSLContext]): SSL context for TLS/SSL connections.
163        tls_params (Optional[TLSParameters]): TLS parameters for secure connections.
164        tls_insecure (Optional[bool]): Allow insecure TLS/SSL connections. Defaults to True.
165        proxy (Optional[ProxySettings]): Proxy configuration settings.
166        socket_options (Optional[Iterable[SocketOption]]): Socket configuration options.
167        session_expiry (int): Session expiry interval in seconds. Defaults to 0.
168        receive_maximum (int): Maximum number of received messages before an ACK (except QoS 0).
169            Defaults to 10.
170        maximum_packet_size (int): Maximum packet size in bytes. Defaults to 4096.
171        topic_alias_maximum (int): Maximum number of topic aliases. Defaults to 0.
172        publish_topics_params (Optional[list[PublishTopicParams]]): List of publish
173            topic parameters.
174        cache_dir (Optional[str]): Directory path for caching.
175        logger (Logger): Logger instance for the class.
176    """

365    async def __anext__(self) -> ItsMessage:
366        """Retrieve and process the next message from the asynchronous message iterator.
367        This method fetches the next message from the underlying message iterator,
368        attempts to deserialize its payload, and constructs an ItsMessage object. If
369        deserialization is successful, the returned ItsMessage will contain the
370        deserialized payload in its_payload. If an error occurs during deserialization,
371        the error is logged and the message is returned with its original payload.
372
373        Returns:
374            ItsMessage: A message container with topic, payload, QoS, retain flag, message ID,
375                        properties, and the deserialized ITS payload if successful.
376
377        Raises:
378            StopAsyncIteration: Implicitly when the underlying iterator is exhausted.
379
380        Note:
381            This method catches all exceptions during deserialization to ensure the iterator
382            continues functioning even if individual messages cannot be properly deserialized.
383        """
384        its_message: Optional[ItsMessage] = None
385        try:
386            while True:
387                message = await anext(self._message_iterator)
388                if not isinstance(message.payload, bytes):
389                    raise StepLibException(
390                        f"Message payload is not bytes: {type(message.payload)}"
391                    )
392                (_, btp_message, facility_raw_message) = (
393                    self._message_serializer.deserialize(message=message.payload)
394                )
395                facility_message = self._message_serializer.convert_to_facility_message(
396                    raw_message=facility_raw_message,
397                    message_type=self._message_serializer.btp_port_to_facility_message_id(
398                        port=BtpWellKnownPort(btp_message.destination_port)
399                    ),
400                )
401                try:
402                    its_message = ItsMessage(
403                        topic=message.topic,
404                        payload=message.payload,
405                        qos=message.qos,
406                        retain=message.retain,
407                        mid=message.mid,
408                        properties=message.properties,
409                        its_payload=facility_message,
410                    )
411                    break
412                except Exception as e:  # pylint: disable=broad-except
413                    self._logger.info(
414                        msg=f"Error during deserializing message: {e}, message {message.payload.hex()}",  # pylint: disable=line-too-long
415                    )
416        except StepLibException as e:
417            self._logger.warning(msg=e)
418        except Exception as e:  # pylint: disable=broad-exception-caught
419            if message.payload is not None:
420                if isinstance(message.payload, str) or isinstance(message.payload, int):
421                    self._logger.warning(
422                        msg=f"Error during deserializing message: {e}, message {message.payload}",
423                        exc_info=e,
424                    )
425                else:
426                    self._logger.warning(
427                        msg=f"Error during deserializing message: {e}, message {message.payload.hex()}",  # pylint: disable=line-too-long
428                        exc_info=e,
429                    )
430            else:
431                self._logger.warning(
432                    msg=f"Error during deserializing message: {e}, message none",
433                    exc_info=e,
434                )
435        if its_message is None:
436            raise StopAsyncIteration
437        return its_message

547    async def publish_its_message(
548        self,
549        message: FacilityMessage,
550        address: GeoNetworkAddress,
551        current_position: GpsLocation,
552        is_station_mobile: bool,
553    ) -> None:
554        """Publishes an ITS message to the appropriate topic based on the message type
555        and geohash precision.
556
557        This method determines the publishing topic based on the message type,
558        creates a geohash from the destination position, and publishes the
559        serialized message to the resulting topic.
560
561        Args:
562            message (FacilityMessage): The ITS message to publish.
563            address (GeoNetworkAddress): The network address information.
564            current_position (GpsLocation): The current GPS position of the sender.
565            is_station_mobile (bool): Flag indicating whether the station is mobile.
566
567        Raises:
568            StepLibException: If no topic configuration is found for the given message type.
569
570        Returns:
571            None
572        """
573        if self._publish_topics_param is None:
574            self._logger.warning(msg="No publish topics configured")
575            return
576        topic_param: PublishTopicParams
577        try:
578            topic_param = next(
579                (
580                    x
581                    for x in self._publish_topics_param
582                    if x.message_type == message.message_type
583                )
584            )
585        except StopIteration as e:
586            raise StepLibException(
587                f"Topic not found for type {message.message_type}"
588            ) from e
589        destination_position = message.reference_position
590        geohash = pygeohash_encode(
591            latitude=destination_position.latitude,
592            longitude=destination_position.longitude,
593            precision=topic_param.geohash_precision,
594        )
595        topic = (
596            f"{topic_param.header_topic}/g{topic_param.geohash_precision}/"
597            + "/".join(geohash)
598        )
599        self._logger.info(msg=f"Publish: topic {topic} message {message}")
600        await self.publish(
601            topic=topic,
602            payload=self._message_serializer.serialize(
603                facility_message=message,
604                address=address,
605                current_position=current_position,
606                is_station_mobile=is_station_mobile,
607            ),
608            qos=topic_param.qos,
609        )
