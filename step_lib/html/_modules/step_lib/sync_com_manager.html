  1"""Synchronous MQTT Communication Manager Module.
  2
  3This module provides a thread-safe interface for MQTT communication with automatic reconnection
  4and message queueing capabilities. It wraps asynchronous MQTT operations in a synchronous
  5interface making it easier to use in traditional threaded applications.
  6
  7Key Features:
  8
  9    - Thread-safe MQTT operations
 10    - Automatic reconnection handling
 11    - Message queueing for incoming and outgoing messages
 12    - Support for QoS levels
 13    - Configurable connection parameters
 14    - Context manager support
 15    - Proper cleanup of resources
 16
 17Example:
 18    >>> params = SyncComManagerParams.get_default()
 19    ... with SyncComManager(params) as manager:
 20    ...     manager.subscribe(['topic1'], MqttQoS.AT_LEAST_ONCE)
 21    ...     manager.publish_its_message(message=some_message, is_station_mobile=True)
 22    ...     received = manager.get_its_message_received()
 23"""

145class SyncComManager:
146    """Synchronous Communication Manager for MQTT messaging. This class provides a
147    thread-safe interface for MQTT communication, managing operations through queues and
148    background threads.
149
150    The manager handles:
151
152        - MQTT message publishing and subscribing
153        - Automatic reconnection on connection loss
154        - Thread-safe message queuing
155        - Exception handling and logging
156
157    Attributes:
158        params (SyncComManagerParams): Configuration parameters for the communication manager,
159            containing all settings (e.g. maximum queue sizes).
160
161    The class can be used either with context manager (recommended) or manual start/stop:
162
163    Example:
164        >>> with SyncComManager(params) as manager:
165        ...    manager.subscribe(['topic1', 'topic2'], QoS.AT_LEAST_ONCE)
166        ...    manager.publish_its_message(message=message, is_station_mobile=True)
167
168    Or manually:
169
170    Example:
171        >>> manager = SyncComManager(params)
172        ... manager.start()
173        ... manager.subscribe(['topic1'], QoS.AT_MOST_ONCE)
174        ... manager.stop()
175    """

322    async def _message_receive_thread(self, client: AsyncComManager) -> None:
323        """Handle receiving messages in a separate thread.
324
325        This method continuously listens for incoming messages from an AsyncComManager client,
326        processes them and puts them into the outgoing queue. It logs the received messages
327        along with their QoS level and topic.

329        Args:
330            client (AsyncComManager): The async communication manager client to receive messages
331            from.

365    async def _unsubscribe_topics(self, client: AsyncComManager) -> None:
366        """Unsubscribes the client from all previously subscribed topics.
367
368        This method removes all topic subscriptions for the given client and clears
369        the internal topics tracking. It logs debug and info messages during the process.

371        Args:
372            client (AsyncComManager): The communication manager client to unsubscribe.
373
374        Returns:
375            None
376
377        Note:
378            - If no topics were previously subscribed, the method returns immediately
379            - After unsubscribing, the internal topics tracking is cleared regardless of success
380        """

392    async def _subscribe_command_exec(
393        self, client: AsyncComManager, subscribe_params: SubscribeCommand
394    ) -> None:
395        """Subscribe to MQTT topics with given parameters.
396
397        This method handles the subscription process to MQTT topics. It first unsubscribes from
398        any existing topics and then subscribes to the new ones specified in subscribe_params.
399
400        Args:
401            client (AsyncComManager): The MQTT client instance used for subscription.
402            subscribe_params (SubscribeCommand): Contains subscription parameters including:
403
404                - topics: List of topics to subscribe to
405                - max_qos: Maximum Quality of Service level for the subscription
406
407        Returns:
408            None
409
410        Note:
411            - The method always updates self._subscribe_topics with the new subscribe_params
412            - Subscription options are set to:
413
414                - noLocal=True (prevent receiving own messages)
415                - retainAsPublished=False
416                - retainHandling=RETAIN_DO_NOT_SEND
417
418            - Logs debug information for command receipt
419            - Logs info for successful subscriptions
420            - Logs errors for failed subscriptions
421        """
422        try:
423            self._params.logger.debug(
424                msg=f"Command received: subscribe {subscribe_params}"
425            )
426            await self._unsubscribe_topics(client=client)
427            if len(subscribe_params.topics) > 0:
428                sub_results = await client.subscribe(
429                    topic=[
430                        (
431                            topic,
432                            SubscribeOptions(
433                                qos=subscribe_params.max_qos,
434                                noLocal=subscribe_params.no_local,
435                                retainAsPublished=False,
436                                retainHandling=SubscribeOptions.RETAIN_DO_NOT_SEND,  # pylint: disable=line-too-long
437                            ),
438                        )
439                        for topic in subscribe_params.topics
440                    ],
441                )
442                for topic, result in zip(
443                    subscribe_params.topics,
444                    sub_results,
445                ):
446                    if result == subscribe_params.max_qos:
447                        self._params.logger.info(msg=f"Subscribed to {topic}: {result}")
448                    else:
449                        self._params.logger.error(
450                            msg=f"Error subscribing to {topic}: {result}"
451                        )
452        finally:
453            self._subscribe_topics = subscribe_params

479    async def _message_publish_thread(self, client: AsyncComManager) -> None:
480        """Executes an asynchronous message publishing thread that handles incoming
481        commands.
482
483        This thread manages subscribing to topics, publishing messages, and handling exit commands.
484        It continuously monitors an incoming queue for commands and processes them accordingly.
485
486        Args:
487            client (AsyncComManager): The asynchronous communication manager client instance.
488
489        Returns:
490            None
491
492        Note:
493            The thread runs continuously until an ExitThreadsCommand is received.
494            Supported commands are:
495
496                - ExitThreadsCommand: Terminates the thread
497                - SubscribeCommand: Subscribes to specified topics
498                - PublishMessageCommand: Publishes a message
499        """
500        if self._subscribe_topics is not None:
501            await self._subscribe_command_exec(
502                client=client, subscribe_params=self._subscribe_topics
503            )
504        loop_continue = True
505        while loop_continue:
506            # Wait for a message
507            msg: SyncComManager.Command = await self._incoming_queue.get()
508            # Process the message
509            match msg:
510                case SyncComManager.ExitThreadsCommand():
511                    await self._exit_threads()
512                    loop_continue = False
513                case SyncComManager.SubscribeCommand():
514                    await self._subscribe_command_exec(
515                        client=client, subscribe_params=msg
516                    )
517                case SyncComManager.PublishMessageCommand():
518                    await self._publish_command_exec(client=client, message=msg)
519                case _:
520                    self._params.logger.warning(
521                        msg=f"Invalid message type: {type(msg)}"
522                    )

545    async def _async_run(self) -> None:
546        """Run asynchronous communication manager.
547
548        This method initializes and manages asynchronous communication threads for message
549        publishing and receiving. It handles connection retries and exceptions during execution.
550
551        The method sets up an event loop with a custom exception handler and creates tasks for:
552
553            - Message receiving (if max_outgoing_queue_rooms > 0)
554            - Message publishing
555
556        The execution continues until either loop_continue is False or _async_thread_enable is
557        False. On exceptions, it will attempt to reconnect after the configured reconnect interval.
558
559        Returns:
560            None
561        """
562        self._set_connection_status(status=ConnectionStatus.DISCONNECTED)
563        self._async_loop = asyncio_get_event_loop()
564        self._async_loop.set_exception_handler(
565            handler=lambda loop, context: self._exception_handler(context=context)
566        )
567        loop_continue = True
568        while loop_continue and self._async_thread_enable:
569            try:
570                async with AsyncComManager(params=self._params) as client:
571                    self._set_connection_status(ConnectionStatus.CONNECTED)
572                    async with AsyncioTaskGroup() as group:
573                        if self._params.max_outgoing_queue_rooms > 0:
574                            # Create mqtt receive thread
575                            self._async_threads.append(
576                                group.create_task(
577                                    coro=self._message_receive_thread(client=client),
578                                    name="message_receive_thread",
579                                )
580                            )
581                        # Create mqtt send thread
582                        self._async_threads.append(
583                            group.create_task(
584                                coro=self._message_publish_thread(client=client),
585                                name="message_publish_thread",
586                            )
587                        )
588                self._set_connection_status(status=ConnectionStatus.DISCONNECTED)
589                loop_continue = False
590            except Exception as e:  # pylint: disable=broad-exception-caught
591                self._set_connection_status(ConnectionStatus.DISCONNECTED)
592                self._params.logger.error(
593                    msg=f"Exception happened: {e}",
594                    exc_info=e,
595                )
596                self._params.logger.info(
597                    msg="All threads stopped and connection dropped"
598                )
599                await asyncio_sleep(delay=self._params.reconnect_interval)
600                self._params.logger.info(
601                    msg=f"Connection retry after {self._params.reconnect_interval}"
602                )
603                self._set_connection_status(ConnectionStatus.RECONNECTING)
604            finally:
605                self._async_threads.clear()
606        self._params.logger.info(msg="Thread async execution end")

641    def stop(self) -> None:
642        """Stop the asynchronous communication thread. Stops the running asynchronous
643        thread by setting the enable flag to False and sending an exit command through
644        the queue. Waits for thread completion before clearing the async loop reference.

646        Raises:
647            StepLibException: If async loop is not running or async thread is not alive.

663        self._async_loop = None
