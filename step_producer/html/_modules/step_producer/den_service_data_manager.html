  1"""DenServiceDataManager Module.
  2
  3This module provides the `DenServiceDataManager` class for managing
  4Decentralized Environmental Notification Message (DENM) data related
  5to collision risk events. It is designed for use in Intelligent
  6Transport Systems (ITS) and handles the creation, storage, updating, and
  7expiration of DENM records. The manager maintains efficient indexes for
  8quick retrieval and manipulation of DENM records, supports thread-safe
  9operations, and integrates with logging for debugging and monitoring.
 10
 11Key Features:
 12
 13    - Creation and update of DENM records for collision scenarios
 14    - Indexing for fast lookup by station and action identifiers
 15    - Expiration and cleanup of outdated DENM records
 16    - Thread-safe operations using internal locking
 17    - Integration with ITS data models and logging

 46class DenServiceDataManager:
 47    """Manages Decentralized Environmental Notification Message (DENM) data for
 48    collision risk events. This class is responsible for creating, storing, updating and
 49    managing DENM records related to potential collisions. It maintains various indexes
 50    for efficient retrieval and processing of DENM records. DenServiceDataManager
 51    organizes collision data into structured records, handles the lifecycle of these
 52    records including creation, update and expiration, and provides methods to access
 53    and manipulate the stored data.
 54
 55    Attributes:
 56        station_id: The ID of the station generating these DENM messages.
 57        time_after_collision: Time to continue broadcasting a DENM after collision (in ms).
 58        denm_last_update_max_delay_ms: Maximum allowed delay for DENM record updates.
 59        denm_check_last_update_max_interval: Maximum interval for checking record updates.
 60        logger: Logger instance for recording operations and debugging information.

 95    def _station_order(self, stations: list[PositionInfo]) -> list[PositionInfo]:
 96        """Sorts a list of stations by their station ID and object ID.
 97
 98        Args:
 99            stations: A list of PositionInfo objects representing stations.
100
101        Returns:
102            A sorted list of PositionInfo objects, ordered by stationId and objectId.
103        """
104        stations = list(stations)
105        return sorted(
106            stations,
107            key=lambda station: (
108                station.stationInfo.stationId,
109                station.objectId if station.objectId else 0,
110            ),
111        )

114        self,
115        record_key: tuple[int, int, int, int],
116        action_id: ActionId,
117        detection_time_its: ItsTime,
118        station_collision_array: StationCollisionArrayV1,
119        position_reference: PositionInfo,
120        station: PositionInfo,
121    ) -> DenmRecord:
122        """Creates a Decentralized Environmental Notification Message (DENM) record for
123        collision risk events.
124
125        This method constructs a DENM record containing information about a potential collision
126        between vehicles. The DENM includes management data, situation information, location
127        details, and pre-crash information.
128
129        Args:
130            record_key (tuple[int, int, int, int]): Unique identifier for the record
131            action_id (ActionId): Action identifier for the DENM
132            detection_time_its (ItsTime): Detection time in ITS format
133            station_collision_array (StationCollisionArrayV1): Information about the
134                collision scenario
135            position_reference (PositionInfo): Position information of the reference station
136            station (PositionInfo): Position information of the other involved station
137
138        Returns:
139            DenmRecord: A complete DENM record containing all necessary information about the
140            potential collision event
141        """
142        return DenmRecord(
143            denm=DenmDataV1(
144                stationId=self._station_id,
145                managementContainer=ManagementContainer(
146                    actionId=action_id,
147                    detectionTime=station_collision_array.detectionTime,
148                    referenceTime=station_collision_array.detectionTime,
149                    termination=None,
150                    eventPosition=position_reference.position,
151                    awarenessDistance=None,
152                    trafficDirection=TrafficDirection.ALL_TRAFFIC_DIR,
153                    validityDuration=station_collision_array.timeToCollision
154                    + self._time_after_collision,
155                    transmissionInterval=None,
156                    stationType=position_reference.stationInfo.stationType,
157                ),
158                situationContainer=SituationContainer(
159                    informationQuality=station_collision_array.informationQuality,
160                    eventType=CauseCode.COLLISION_RISK,
161                ),
162                locationContainer=LocationContainer(
163                    eventSpeed=position_reference.speed,
164                    eventPositionHeading=position_reference.heading,
165                    roadType=station_collision_array.roadType,
166                ),
167                preCrashContainer=PreCrashContainer(
168                    perceivedObject=PerceivedObject(
169                        objectId=station.objectId,
170                        measurementDeltaTime=ItsTime.create_from_iso_time(
171                            time_str=station.time
172                        ).its_ms
173                        - detection_time_its.its_ms,
174                        position=station.position,
175                        speed=station.speed,
176                        heading=station.heading,
177                        objectDimensionZ=None,
178                        objectDimensionY=None,
179                        objectDimensionX=None,
180                        objectPerceptionQuality=None,
181                    ),
182                    stationIdInvolved=station.stationInfo.stationId,
183                    timeToCollision=station_collision_array.timeToCollision,
184                    impactSection=None,
185                    estimatedBrakingDistance=station_collision_array.estimatedBrakingDistance,
186                ),
187            ),
188            record_key=record_key,
189        )

191    def insert_station_collision_array(
192        self, station_collision_array: StationCollisionArrayV1
193    ) -> list[DenmRecord]:
194        """Processes a StationCollisionArrayV1 and creates or updates DENM records for
195        station pairs.
196
197        This method takes a StationCollisionArrayV1 object, orders the stations it contains,
198        and creates DENM (Decentralized Environmental Notification Message) records for each
199        pair of stations where the first station is used as a position reference.
200
201        Args:
202            station_collision_array (StationCollisionArrayV1): The collision array containing
203                stations information and detection time.
204
205        Returns:
206            list[DenmRecord]: A list of new or updated DENM records created during processing.
207
208        Note:
209            - For each pair of stations, a unique record key is created based on station IDs and
210              object IDs.
211            - If a record with the same key already exists, its action ID is reused, otherwise a
212              new one is created.
213            - The sequence number is incremented for each new action ID.
214            - The method maintains several indexes for efficient record retrieval and management.
215            - The reference of the DENM is always the first station in the ordered list.
216            - This method is thread-safe and uses internal locking to ensure data consistency
217              during the cleanup operation.
218        """
219        detection_time_its = ItsTime.create_from_iso_time(
220            time_str=station_collision_array.detectionTime
221        )
222        ordered_stations = self._station_order(
223            stations=station_collision_array.stations
224        )
225        position_reference: Optional[PositionInfo] = None
226        denm_list: list[DenmRecord] = []
227        with self._lock:
228            for station in ordered_stations:
229                self._logger.debug(
230                    msg=f"Processing station: {station.stationInfo.stationId}, objectId: {station.objectId}"  # pylint: disable=line-too-long
231                )
232                if position_reference is None:
233                    position_reference = station
234                else:
235                    # Create a unique record key based on the involved stations
236                    record_key = (
237                        position_reference.stationInfo.stationId,
238                        (
239                            position_reference.objectId
240                            if position_reference.objectId
241                            else 0
242                        ),
243                        station.stationInfo.stationId,
244                        station.objectId if station.objectId else 0,
245                    )
246                    self._logger.debug(msg=f"record key: {record_key}")
247                    # Create or retrieve the action_id
248                    action_id: ActionId
249                    if record_key in self._record_key_index:
250                        action_id = self._record_key_index[
251                            record_key
252                        ].denm.managementContainer.actionId
253                        self._logger.debug(
254                            msg=f"Found existing record {record_key} action id: {action_id}"
255                        )
256                        self._denm_sorted_records.discard(
257                            value=self._record_key_index[record_key]
258                        )
259                    else:
260                        action_id = ActionId(
261                            originatingStationId=self._station_id,
262                            sequenceNumber=self._sequence_number,
263                        )
264                        self._logger.debug(
265                            msg=f"New record {record_key} action id: {action_id}"
266                        )
267                        # Wrap around at 65536
268                        self._sequence_number = (self._sequence_number + 1) % 65536
269                    # Create the new DENM record
270                    record = self._create_denm(
271                        record_key=record_key,
272                        action_id=action_id,
273                        detection_time_its=detection_time_its,
274                        station_collision_array=station_collision_array,
275                        position_reference=position_reference,
276                        station=station,
277                    )
278                    # Create or update the DENM record
279                    self._record_key_index[record_key] = record
280                    self._action_id_index[
281                        (action_id.originatingStationId, action_id.sequenceNumber)
282                    ] = record
283                    self._denm_sorted_records.add(value=record)
284                    denm_list.append(record)
285        return denm_list

287    def get_denm_records_clean_reference_expired(
288        self,
289    ) -> tuple[list[DenmRecord], Optional[DenmRecord]]:
290        """Remove expired DENM records and those with outdated reference times.
291
292        This method performs cleanup of DENM (Decentralized Environmental Notification Message)
293        records by removing those that have either expired based on their validity duration
294        or have reference times older than the maximum allowed delay.
295
296        The method processes records in chronological order (oldest first) and:
297
298            #. Removes records that have exceeded their validity duration
299            #. Removes records with reference times older than the minimum threshold
300            #. Stops processing when it encounters a valid, non-expired record
301            #. Updates internal indexes and logs removal operations
302
303        Returns:
304            tuple[list[DenmRecord], Optional[DenmRecord]]: A tuple containing:
305
306                - A list of removed DENM records that were expired or had outdated reference times
307                - The oldest remaining valid record, or None if no valid records remain
308
309        Note:
310            This method is thread-safe and uses internal locking to ensure data consistency
311            during the cleanup operation.
312        """
313        now = ItsTime()
314        reference_time_min = now - self._denm_last_update_max_delay_ms
315        records_to_remove: list[DenmRecord] = []
316        with self._lock:
317            while True:
318                oldest_record: Optional[DenmRecord]
319                try:
320                    oldest_record = cast(
321                        DenmRecord, self._denm_sorted_records.pop(index=0)
322                    )
323                except IndexError:
324                    oldest_record = None
325                    break
326                validity_duration = (
327                    oldest_record.denm.managementContainer.validityDuration
328                    if oldest_record.denm.managementContainer.validityDuration
329                    else 1000
330                )
331                if now <= oldest_record.reference_time + validity_duration:
332                    if oldest_record.reference_time >= reference_time_min:
333                        self._denm_sorted_records.add(value=oldest_record)
334                        break
335                    records_to_remove.append(oldest_record)
336                del self._record_key_index[oldest_record.record_key]
337                del self._action_id_index[oldest_record.action_id_index]
338                self._logger.debug(
339                    msg=f"Removing DENM: key {oldest_record.record_key}, "
340                    f"action ID {oldest_record.action_id_index}, "
341                    f"reference time {oldest_record.reference_time} ms"
342                )
343        return (records_to_remove, oldest_record)

345    def get_interval_oldest_update_ms(
346        self, oldest_record: Optional[DenmRecord] = None
347    ) -> int:
348        """Calculates the time interval in milliseconds since the oldest DENM record was
349        updated.
350
351        If an `oldest_record` is provided, the interval is calculated from its reference time.
352        Otherwise, the interval is calculated from the reference time of the first record in
353        the `_denm_sorted_records` list. If no records are available, returns the maximum
354        allowed interval (`_denm_check_last_update_max_interval`).
355
356        Args:
357            oldest_record (Optional[DenmRecord]): The DENM record to use as the oldest record.
358                If None, the method uses the first record in `_denm_sorted_records`.
359
360        Returns:
361            int: The interval in milliseconds since the oldest DENM record was updated.
362
363        Note:
364            This method is thread-safe and uses internal locking to ensure data consistency
365            during the cleanup operation.
366        """
367        with self._lock:
368            try:
369                if oldest_record is not None:
370                    record = oldest_record
371                else:
372                    record = cast(DenmRecord, self._denm_sorted_records[0])
373                interval = ItsTime().its_ms - record.reference_time.its_ms
374            except IndexError:
375                self._logger.debug(msg="No DENM records found")
376                interval = self._denm_check_last_update_max_interval
377            if interval > self._denm_check_last_update_max_interval:
378                self._logger.debug(
379                    msg=f"Oldest DENM record update interval {interval} ms "
380                    f"exceeded maximum {self._denm_check_last_update_max_interval} ms"
381                )
382                interval = self._denm_check_last_update_max_interval
383        self._logger.debug(msg=f"Oldest DENM record update interval: {interval} ms")
384        return interval
