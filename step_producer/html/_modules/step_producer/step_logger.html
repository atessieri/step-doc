 37class UtcFormatter(Formatter):
 38    """This class definition creates a custom logging formatter called UtcFormatter that
 39    inherits from Python's built-in logging.Formatter class.
 40
 41    Here's what the class method does:
 42    converter: This class attribute sets the time conversion function to time.gmtime,
 43    which converts a time in seconds since the epoch to a struct_time in UTC
 44    (Coordinated Universal Time) zone.
 45    This means that any timestamps generated by this formatter will be in UTC.
 46    Note that this class doesn't define any instance methods, only a class attribute.
 47    """
 48
 49    __slots__: list[str] = []

 64def _rotator(source: str, dest: str) -> None:
 65    """Compresses a file using BZIP2 compression and saves it to the specified
 66    destination, then removes the original file.
 67
 68    Args:
 69        source (str): The path to the source file to be compressed.
 70        dest (str): The path where the compressed file will be saved.
 71
 72    Returns:
 73        None
 74    """
 75    with ZipFile(
 76        file=dest,
 77        mode="w",
 78        compression=zipfile.ZIP_BZIP2,
 79        allowZip64=True,
 80        compresslevel=9,
 81    ) as zip_file:
 82        zip_file.write(
 83            filename=source,
 84            arcname=basename(p=source),
 85        )
 86        remove(path=source)

101def get_logger(config: Optional[LogConfig]) -> Logger:
102    """Configure and return a logger based on the provided configuration.
103
104    This function creates and configures a logger with specific formatting and handling based on
105    the input configuration. If no configuration is provided, default settings are used.
106
107    The logger can be configured with different handlers based on the config:
108
109        - StreamHandler (default): Logs to stdout
110        - NullHandler: Suppresses all logging
111        - FileHandler: Logs to a single file
112        - RotatingFileHandler: Size-based log rotation
113        - TimedRotatingFileHandler: Time-based log rotation
114
115    The log format, date format, and log level are also configurable through the config object.
116    When no config is provided, logs are set to INFO level with a default UTC formatted output
117    to stdout.
118
119    Args:
120        config (Optional[LogConfig]): Configuration object containing
121            logging settings. If None, default settings will be used.
122
123    Returns:
124        Logger: Configured logger instance with appropriate handler and formatter.
125    """
126    logger = getLogger(name=__name__)
127    logger.handlers.clear()
128    handler: Handler
129    if config is None:
130        logger.setLevel(level=logging.INFO)
131        handler = StreamHandler(stream=stdout)
132        formatter = UtcFormatter(
133            fmt="%(asctime)s.%(msecs)03d [%(levelname)s][pid=%(process)d][thread=%(threadName)s][module=%(module)s][function=%(funcName)s]: %(message)s",  # pylint: disable=line-too-long
134            datefmt="%Y-%m-%d %H:%M:%S",
135        )
136    else:
137        logger.setLevel(level=LogMode.to_logging_level(mode=config.level))
138        match config.handler:
139            case LogStreamHandler():
140                handler = StreamHandler(stream=stdout)
141            case LogNullHandler():
142                handler = NullHandler()
143            case LogFileHandler():
144                makedirs(
145                    name=dirname(p=config.handler.filename),
146                    exist_ok=True,
147                )
148                handler = FileHandler(
149                    filename=config.handler.filename,
150                    mode="a" if config.handler.append else "w",
151                    encoding="utf-8",
152                    delay=False,
153                    errors=None,
154                )
155            case LogSizeFileRotateHandler():
156                makedirs(
157                    name=dirname(p=config.handler.filename),
158                    exist_ok=True,
159                )
160                handler = RotatingFileHandler(
161                    filename=config.handler.filename,
162                    maxBytes=config.handler.maxBytes,
163                    backupCount=config.handler.backupCount,
164                    encoding="utf-8",
165                    delay=False,
166                    errors=None,
167                )
168                handler.rotator = _rotator
169                handler.namer = _namer
170            case LogTimeFileRotateHandler():
171                makedirs(
172                    name=dirname(p=config.handler.filename),
173                    exist_ok=True,
174                )
175                handler = TimedRotatingFileHandler(
176                    filename=config.handler.filename,
177                    when=LogIntervalType.to_when(
178                        interval_type=config.handler.intervalType
179                    ),
180                    interval=config.handler.interval,
181                    backupCount=config.handler.backupCount,
182                    encoding="utf-8",
183                    delay=False,
184                    utc=True,
185                )
186                handler.rotator = _rotator
187                handler.namer = _namer
188            case _:
189                handler = StreamHandler(stream=stdout)
190        formatter = UtcFormatter(
191            fmt=config.format,
192            datefmt=config.dateFormat,
193        )
194    handler.setFormatter(fmt=formatter)
195    logger.addHandler(hdlr=handler)
196    return logger
