  1"""STEP Producer Redis Integration Module.
  2
  3This module provides specialized Redis integration for STEP (Secure Transport Events Platform)
  4Producer services. It defines Redis connection pools and streams optimized for handling ITS
  5(Intelligent Transportation Systems) messages, with a focus on DENM (Decentralized Environmental
  6Notification Message) data.
  7
  8Key Components:
  9
 10    #. StepProducerRedisPool:
 11       A specialized Redis connection pool for STEP Producers that limits the maximum number of
 12       connections. Designed to be used as an async context manager for efficient resource
 13       management.
 14
 15    #. StepProducerRedisStream:
 16       A Redis stream implementation tailored for STEP Producer service that handles:
 17
 18        - Creating and managing client input streams
 19        - Publishing ITS messages (DENM data) to client streams
 20        - Receiving and validating ITS messages from Redis streams
 21
 22The module integrates with the STEP ecosystem by leveraging the step_cloud_schemas for message
 23structures and step_redis for base Redis functionality, providing a seamless interface for
 24STEP Producers to interact with Redis for real-time ITS message processing.
 25
 26Example Usage:
 27    >>> # Set up Redis connection and stream
 28    ... async with StepProducerRedisPool(config, logger) as redis_pool:
 29    ...     stream = StepProducerRedisStream(consumer_name, stream_config, logger, redis_pool)
 30    ...     async with stream:
 31    ...         # Create input stream
 32    ...         await stream.create_client_input_stream()
 33    ...
 34    ...         # Receive and process messages
 35    ...         message = await stream.receive_its_message()
 36    ...
 37    ...         # Publish processed messages
 38    ...         await stream.publish_its_message(denm_data)
 39"""

132class StepProducerRedisStream(StepRedisStream):
133    """A specialized Redis stream implementation for the Step Producer service. This
134    class extends the StepRedisStream base class to provide functionality specific to
135    Step Producer needs, such as consuming ITS (Intelligent Transportation System)
136    messages and publishing DENM (Decentralized Environmental Notification Message) data
137    to client streams. The class implements asynchronous context management and
138    specialized stream handling methods for ITS message processing workflows.
139
140    Attributes:
141        consumer_name (str): The name of the consumer utilizing this producer.
142        config (RedisStreams): Configuration for Redis streams.
143        logger (Logger): Logger instance for logging operations.
144        pool (StepRedisPool): Redis connection pool for step operations.
145    """

194    async def publish_its_message(self, message: DenmDataV1) -> None:
195        """Publishes a DENM message to the step client stream.
196
197        This method converts the DENM message to a CloudEvents JSON format
198        expected by the client, then publishes it to the configured output stream.
199
200        Args:
201            message (DenmDataV1): The DENM (Decentralized Environmental Notification Message)
202                data to be published.
203
204        Returns:
205            None
206        """
207        message_step_client = ClientCloudEventsJsonRoot(
208            root=ClientCloudEventsDenmDataV1(
209                id=str(object=uuid4()),
210                source=self._consumer_name,
211                specversion=SpecVersion.V1_0,
212                type=ClientMessageType.DENM,
213                dataschema=ClientDataSchemaType.DENM_DATA_V1_SCHEMA,
214                datacontenttype=DataContentType.APPLICATION_JSON,
215                time=str(object=message.managementContainer.referenceTime),
216                data=message,
217            )
218        ).model_dump_json()
219        self._logger.info(msg=f"Publish message to step client: {message_step_client}")
220        await self.publish_message(
221            stream_id=self._config.outputs.client.streamId,
222            message=message_step_client,
223            stream_max_len=self._config.outputs.client.maxLength,
224        )
