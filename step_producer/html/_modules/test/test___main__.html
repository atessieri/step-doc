120def test___os_signal_config(mocker: MockerFixture) -> None:
121    """Test the _os_signal_config function for both Windows and Unix-like platforms.
122
123    This test verifies that the function correctly configures signal handlers
124    based on the operating system platform, using either signal.signal() for
125    Windows or asyncio.get_running_loop().add_signal_handler() for Unix-like systems.
126
127    Args:
128        mocker: MockerFixture for creating mock objects during testing.
129
130    Returns:
131        None
132    """
133    # Arrange
134    mock_valid_signal = mocker.MagicMock(spec=Signals)
135    mock_valid_signal.name = "SIGTERM"
136    mock_valid_signal.value = 15
137    mocker.patch(
138        "step_producer.__main__.valid_signals", return_value=[mock_valid_signal]
139    )
140    # mocker.patch("step_producer.__main__.sys.platform", platform)
141    # mocker.patch("step_producer.__main__.os.name", os_name)
142    mock_signal_func = mocker.patch("step_producer.__main__.signal", return_value=None)
143    mock_loop = mocker.MagicMock()
144    mock_loop.add_signal_handler = mocker.MagicMock()
145    mock_get_running_loop = mocker.patch(
146        "step_producer.__main__.get_running_loop",
147        return_value=mock_loop,
148    )
149    # Act
150    _os_signal_config()
151    # Assert
152    if sys.platform.lower() == "win32" or os.name.lower() == "nt":
153        # On Windows, should use signal.signal()
154        mock_signal_func.assert_called_once_with(
155            signalnum=mock_valid_signal, handler=mocker.ANY
156        )
157        mock_get_running_loop.assert_not_called()
158        mock_loop.add_signal_handler.assert_not_called()
159    else:
160        # On Unix-like systems, should use asyncio loop signal handler
161        mock_signal_func.assert_not_called()
162        mock_get_running_loop.assert_called_once()
163        mock_loop.add_signal_handler.assert_called_once_with(
164            sig=mock_valid_signal, callback=mocker.ANY
165        )
