 36class UtcFormatter(Formatter):
 37    """This class definition creates a custom logging formatter called UtcFormatter that
 38    inherits from Python's built-in logging.Formatter class.
 39
 40    Here's what the class method does:
 41    converter: This class attribute sets the time conversion function to time.gmtime,
 42    which converts a time in seconds since the epoch to a struct_time in UTC
 43    (Coordinated Universal Time) zone.
 44    This means that any timestamps generated by this formatter will be in UTC.
 45    Note that this class doesn't define any instance methods, only a class attribute.
 46    """

 53            timestamp (Optional[float]): Seconds since the Epoch. Default: current time.

 55        Returns:
 56            struct_time: A time object in UTC timezone.

 61def _rotator(source: str, dest: str) -> None:
 62    """Compresses a file using BZIP2 compression and saves it to the specified
 63    destination, then removes the original file.
 64
 65    Args:
 66        source (str): The path to the source file to be compressed.
 67        dest (str): The path where the compressed file will be saved.
 68
 69    Returns:
 70        None
 71    """
 72    with ZipFile(
 73        file=dest,
 74        mode="w",
 75        compression=ZIP_BZIP2,
 76        allowZip64=True,
 77        compresslevel=9,
 78    ) as zip_file:
 79        zip_file.write(
 80            filename=source,
 81            arcname=basename(p=source),
 82        )
 83        remove(path=source)

 98def get_logger(config: Optional[LogConfig]) -> Logger:
 99    """Configure and return a logger based on the provided configuration.
100
101    This function creates and configures a logger with specific formatting and handling based on
102    the input configuration. If no configuration is provided, default settings are used.
103
104    The logger can be configured with different handlers based on the config:
105
106        - StreamHandler (default): Logs to stdout
107        - NullHandler: Suppresses all logging
108        - FileHandler: Logs to a single file
109        - RotatingFileHandler: Size-based log rotation
110        - TimedRotatingFileHandler: Time-based log rotation
111
112    The log format, date format, and log level are also configurable through the config object.
113    When no config is provided, logs are set to INFO level with a default UTC formatted output
114    to stdout.
115
116    Args:
117        config (Optional[LogConfig]): Configuration object containing
118            logging settings. If None, default settings will be used.
119
120    Returns:
121        Logger: Configured logger instance with appropriate handler and formatter.
122    """
123    logger = getLogger(name=__name__)
124    logger.handlers.clear()
125    handler: Handler
126    if config is None:
127        logger.setLevel(level=logging.INFO)
128        handler = StreamHandler(stream=stdout)
129        formatter = UtcFormatter(
130            fmt="%(asctime)s.%(msecs)03d [%(levelname)s][pid=%(process)d][thread=%(threadName)s][module=%(module)s][function=%(funcName)s]: %(message)s",  # pylint: disable=line-too-long
131            datefmt="%Y-%m-%d %H:%M:%S",
132        )
133    else:
134        logger.setLevel(level=LogMode.to_logging_level(mode=config.level))
135        match config.handler:
136            case LogStreamHandler():
137                handler = StreamHandler(stream=stdout)
138            case LogNullHandler():
139                handler = NullHandler()
140            case LogFileHandler():
141                makedirs(
142                    name=dirname(p=config.handler.filename),
143                    exist_ok=True,
144                )
145                handler = FileHandler(
146                    filename=config.handler.filename,
147                    mode="a" if config.handler.append else "w",
148                    encoding="utf-8",
149                    delay=False,
150                    errors=None,
151                )
152            case LogSizeFileRotateHandler():
153                makedirs(
154                    name=dirname(p=config.handler.filename),
155                    exist_ok=True,
156                )
157                handler = RotatingFileHandler(
158                    filename=config.handler.filename,
159                    maxBytes=config.handler.maxBytes,
160                    backupCount=config.handler.backupCount,
161                    encoding="utf-8",
162                    delay=False,
163                    errors=None,
164                )
165                handler.rotator = _rotator
166                handler.namer = _namer
167            case LogTimeFileRotateHandler():
168                makedirs(
169                    name=dirname(p=config.handler.filename),
170                    exist_ok=True,
171                )
172                handler = TimedRotatingFileHandler(
173                    filename=config.handler.filename,
174                    when=LogIntervalType.to_when(
175                        interval_type=config.handler.intervalType
176                    ),
177                    interval=config.handler.interval,
178                    backupCount=config.handler.backupCount,
179                    encoding="utf-8",
180                    delay=False,
181                    utc=True,
182                )
183                handler.rotator = _rotator
184                handler.namer = _namer
185            case _:
186                handler = StreamHandler(stream=stdout)
187        formatter = UtcFormatter(
188            fmt=config.format,
189            datefmt=config.dateFormat,
190        )
191    handler.setFormatter(fmt=formatter)
192    logger.addHandler(hdlr=handler)
193    return logger
