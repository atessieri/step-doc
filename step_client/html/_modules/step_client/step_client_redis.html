  1"""STEP Client Redis Integration.
  2
  3This module provides Redis-based communication functionality for the STEP client, enabling
  4interaction with the STEP ecosystem through Redis streams. It implements connection pooling
  5and stream handling specifically tailored for Intelligent Transport Systems (ITS) message
  6exchange between clients, adapters, and producers.
  7
  8The module contains two main classes:
  9
 10    #. StepClientRedisPool: Extends StepRedisPool to provide a Redis connection pool
 11       specifically configured for STEP client needs, handling connection parameters
 12       and management.
 13    #. StepClientRedisStream: Extends StepRedisStream to implement stream handling for
 14       ITS messages (CAM, DENM) according to the CloudEvents specification, including:
 15
 16        - Creating and managing input streams with group IDs
 17        - Publishing messages to adapter and Producer streams
 18        - Receiving and parsing messages from configured streams
 19        - Converting between CloudEvents format and STEP facility messages
 20
 21This module serves as the communication layer for STEP clients, enabling standardized
 22message exchange in ITS applications through Redis infrastructure.
 23"""

113class StepClientRedisStream(StepRedisStream):
114    """A Redis stream client implementation for handling ITS (Intelligent Transportation
115    Systems) messages.
116
117    This class extends StepRedisStream to provide specialized functionality for consuming and
118    producing ITS messages through Redis streams. It manages input streams for receiving messages
119    and output streams for publishing CAM (Cooperative Awareness Messages) and DENM
120    (Decentralized Environmental Notification Messages).
121    The client operates as an asynchronous context manager and handles CloudEvents-formatted
122    messages, providing automatic serialization/deserialization and stream management including
123    TTL (Time To Live) configuration.
124    This class is designed for ITS applications and expects specific message formats
125    (CAM and DENM) conforming to CloudEvents specifications. Stream TTL is automatically
126    managed to prevent memory leaks in Redis.
127
128    Attributes:
129        consumer_name (str): The name of the consumer that will be used to identify this
130            client in Redis streams.
131        config (RedisStreams): Configuration object containing Redis streams settings.
132        logger (Logger): Logger instance for logging client operations.
133        pool (StepRedisPool): Redis connection pool for managing connections.
134    """
135
136    __slots__ = ("_consumer_name", "_config", "_stream_ttl")
137
138    def __init__(
139        self,
140        consumer_name: str,
141        config: RedisStreams,
142        logger: Logger,
143        pool: StepRedisPool,
144    ) -> None:
145        self._consumer_name = consumer_name
146        self._config = config
147        # Convert stream_ttl from milliseconds to seconds
148        self._stream_ttl = (config.input.waitTimeout * 2) // 1000
149        super().__init__(pool=pool, logger=logger)

185    async def publish_cam_message(self, message: CamDataV1) -> None:
186        """Publishes a CAM (Cooperative Awareness Message) to the adapter stream.
187
188        This method wraps the provided CAM message in a CloudEvents format adapter
189        and publishes it to the configured adapter output stream. The message is
190        serialized as JSON and includes metadata such as unique ID, source, timestamp,
191        and data schema information.
192
193        Args:
194            message (CamDataV1): The CAM data message to be published.
195
196        Returns:
197            None
198
199        Note:
200            The method logs the adapter message content before publishing for
201            debugging and monitoring purposes.
202        """
203        message_adapter = AdapterCloudEventsJsonRoot(
204            root=AdapterCloudEventsCamDataV1(
205                id=str(object=uuid4()),
206                source=self._consumer_name,
207                specversion=SpecVersion.V1_0,
208                type=AdapterMessageType.CAM,
209                dataschema=AdapterDataSchemaType.CAM_DATA_V1_SCHEMA,
210                datacontenttype=DataContentType.APPLICATION_JSON,
211                time=message.creationTime,
212                data=message,
213            )
214        ).model_dump_json()
215        self._logger.info(msg=f"Publish CAM to adapter: {message_adapter}")
216        await self.publish_message(
217            stream_id=self._config.outputs.adapter.streamId,
218            message=message_adapter,
219            stream_max_len=self._config.outputs.adapter.maxLength,
220        )

222    async def publish_denm_message(self, message: DenmDataV1) -> None:
223        """Publishes a DENM (Decentralized Environmental Notification Message) to the
224        Producer stream.
225
226        This method wraps the provided DENM message in a CloudEvents format with appropriate
227        metadata and publishes it to the configured Redis stream.
228
229        Args:
230            message (DenmDataV1): The DENM message data to be published. Must contain a
231                managementContainer with referenceTime for time-stamping.
232
233        Returns:
234            None
235
236        Note:
237            The message is automatically assigned a unique ID and timestamped with the
238            reference time from the message's management container.
239        """
240        message_producer = ProducerCloudEventsJsonRoot(
241            root=ProducerCloudEventsDenmDataV1(
