   2"""Point interpolation module for temporal and spatial trajectory smoothing.
   3
   4This module provides a comprehensive solution for interpolating sparse position data
   5into smooth, time-aligned trajectories. It is primarily designed for processing
   6GNSS (Global Navigation Satellite System) data from tracking, navigation
   7systems, and other location-based applications where temporal consistency and
   8smooth motion representation are critical.
   9
  10Key Components:
  11    PointInterpolationConfig: Configuration dataclass for interpolation parameters
  12    PointInterpolation: Main interpolation engine with cubic spline-based smoothing
  13
  14Features:
  15    - **Cubic Spline Interpolation**: Uses scipy CubicSpline for smooth spatial
  16      interpolation across latitude, longitude, and altitude dimensions
  17    - **Temporal Validation**: Enforces configurable minimum/maximum periods between
  18      points to maintain data quality and prevent unrealistic interpolations
  19    - **Motion Parameter Calculation**: Automatically computes speed and heading
  20      from position changes when missing from input data
  21    - **Adaptive Output Generation**: Produces time-aligned sequences at configurable
  22      nominal periods with tolerance for timing variations
  23    - **Gap Detection and Recovery**: Handles large temporal gaps by clearing
  24      interpolation history to prevent artifacts across data discontinuities
  25    - **Memory-Efficient Processing**: Uses deque-based sliding windows for optimal
  26      memory usage in continuous data streams
  27
  28Performance Characteristics:
  29    - **Time Complexity**: O(n + k*log(n) + k) per point addition where n is input
  30      queue size and k is output points generated
  31    - **Space Complexity**: O(n + k) with bounded queues for steady-state operation
  32    - **Memory Efficiency**: Automatic cleanup of outdated data prevents unbounded
  33      memory growth during long-running operations
  34
  35Typical Use Cases:
  36    - Vehicle trajectory smoothing for fleet management systems
  37    - Temporal alignment of multi-source position data
  38    - Generation of uniform sampling rates from irregular GPS updates
  39    - Real-time position interpolation for navigation applications
  40    - Motion analysis and parameter estimation from sparse position data
  41
  42Dependencies:
  43    - scipy: For CubicSpline interpolation algorithms
  44    - numpy: For efficient numerical array operations
  45    - step_cloud_schemas: For standardized position data structures
  46    - collections: For deque-based queue management
  47
  48Thread Safety:
  49    Classes in this module are **not thread-safe**. External synchronization
  50    is required when accessing from multiple threads concurrently.
  51
  52Example:
  53    Basic trajectory interpolation workflow:
  54
  55    >>> from step_adapter.point_interpolation import (
  56    ...     PointInterpolationConfig, PointInterpolation
  57    ... )
  58    >>> import logging
  59    >>>
  60    >>> # Configure interpolation parameters
  61    >>> config = PointInterpolationConfig(
  62    ...     path_length_ms=5000,        # 5-second sliding window
  63    ...     output_nominal_period_ms=100, # 10 Hz output rate
  64    ...     period_max_deviation_ms=20   # 20ms tolerance
  65    ... )
  66    >>>
  67    >>> # Create interpolator instance
  68    >>> logger = logging.getLogger(__name__)
  69    >>> interpolator = PointInterpolation(config, logger)
  70    >>>
  71    >>> # Process incoming position data
  72    >>> for timestamp, position_info in position_stream:
  73    ...     result = interpolator.add_point(timestamp, position_info)
  74    ...     if result == PointInterpolation.InsertPointResult.SUCCESS:
  75    ...         interpolated_positions = interpolator.get_interpolated_values()
  76    ...         # Process interpolated positions...

 126    def __post_init__(self) -> None:
 127        """Post-initialization validation and setup for the dataclass.
 128
 129        Validates that the output nominal period and maximum deviation parameters
 130        are within acceptable ranges, then calculates and sets the minimum period.
 131
 132        Returns:
 133            None
 134
 135        Raises:
 136            ValueError: The exception raised if the configuration parameters are invalid.
 137
 138        Note:
 139            The following conditions will raise a ValueError:
 140
 141                - ``path_length_ms`` is less than or equal to the double of
 142                  ``output_nominal_period_ms``.
 143                - ``output_nominal_period_ms`` is negative or zero.
 144                - ``period_max_deviation_ms`` is negative or greater than or equal to
 145                  ``output_nominal_period_ms``.
 146        """
 147        if (self.output_nominal_period_ms <= 0) or (
 148            self.output_nominal_period_ms > 1000
 149        ):
 150            raise ValueError(
 151                f"Output nominal period {self.output_nominal_period_ms} must be "
 152                "greater than 0 and less than 1000"
 153            )
 154        if self.path_length_ms <= self.output_nominal_period_ms * 2:
 155            raise ValueError(
 156                f"Path length {self.path_length_ms} must be greater than "
 157                f"double the output nominal period {self.output_nominal_period_ms}"
 158            )
 159        if (self.period_max_deviation_ms < 0) or (
 160            self.period_max_deviation_ms >= self.output_nominal_period_ms

 162            raise ValueError(
 163                f"Max deviation {self.period_max_deviation_ms} must be "
 164                "greater than or equal to 0 and less than the nominal period "
 165                f"{self.output_nominal_period_ms}"

 169class PointInterpolation:
 170    """A robust point interpolation system for temporal and spatial trajectory
 171    smoothing.
 172
 173    This class implements a sophisticated interpolation system designed for processing
 174    time-series position data, typically from GNSS devices. It creates smooth
 175    trajectories by interpolating between sparse position points using cubic spline
 176    interpolation, while maintaining temporal consistency and providing comprehensive
 177    motion parameter calculations.
 178
 179    The system operates on a sliding window approach, maintaining both input and output
 180    queues to handle continuous data streams efficiently. It performs temporal validation,
 181    gap detection, and automatic queue management to ensure data quality and prevent
 182    interpolation across unrealistic time gaps.
 183
 184    Key Features:
 185        - **Cubic Spline Interpolation**: Uses scipy cubic splines for smooth spatial
 186          interpolation across latitude, longitude, and altitude dimensions.
 187        - **Temporal Validation**: Enforces minimum and maximum time periods between
 188          points to maintain data quality and prevent unrealistic interpolations.
 189        - **Motion Parameter Calculation**: Automatically computes missing speed and
 190          heading information based on position changes and time differences.
 191        - **Adaptive Output Generation**: Produces time-aligned position sequences at
 192          configurable nominal periods with deviation tolerance.
 193        - **Queue Management**: Efficiently manages sliding windows of input and output
 194          data with automatic cleanup of outdated points.
 195        - **Gap Detection**: Identifies and handles large temporal gaps by clearing
 196          history to prevent interpolation across data discontinuities.
 197
 198    Algorithm Overview:
 199        The interpolation process follows a multi-stage pipeline:
 200
 201            #. **Input Validation**: Validates temporal consistency and rejects invalid points
 202            #. **Queue Management**: Maintains sliding windows of recent position data
 203            #. **Spline Creation**: Generates cubic splines for lat/lon/alt interpolation
 204            #. **Output Generation**: Creates time-aligned position sequences
 205            #. **Motion Calculation**: Computes speed/heading from position differences
 206
 207    Time Complexity Analysis:
 208        **Primary Operation - add_point()**: O(n + k*log(n)) where:
 209
 210            - n = input queue size (typically bounded by path_length_ms / min_period_ms)
 211            - k = number of output points generated per operation
 212
 213        Detailed Breakdown:
 214            - Input validation: O(1) - constant time checks
 215            - Queue cleanup: O(m) where m = outdated points removed (typically small closed
 216              to 1 O(1))
 217            - **Cubic spline creation: O(n)** - linear in input points
 218            - **Spline evaluation: O(k*log(n))** - k evaluations, each O(log(n)) binary search
 219            - **Position object creation: O(k)** - linear in output points
 220            - **Motion calculations: O(k)** - distance/bearing for each output point
 221            - **Queue operations: O(k)** - deque appendleft operations.
 222
 223        Scenarios:
 224            - **Best Case**: O(1): validation failures or insufficient data
 225            - **Average Case**: O(n + k*log(n) + k): successful interpolation (typical case)
 226            - **Worst Case**: O(n + k*log(n) + k + n): dense input requiring extensive cleanup
 227              (corner case)
 228
 229        Auxiliary Operations:
 230            - get_interpolated_values(): O(k) - linear scan of output queue
 231            - Property access (last_point_time): O(1) - constant time
 232            - Clear history: O(1) - deque.clear() is constant time
 233
 234    Space Complexity Analysis:
 235        Overall Space Complexity: O(n + k) where:
 236
 237            - n = maximum input queue size
 238            - k = maximum output queue size (bounded by maxlen parameter)
 239
 240        Detailed Memory Usage:
 241            - **Input Queue**: O(n) space for storing (timestamp, PositionInfo) tuples
 242              Typical size: path_length_ms / output_nominal_period_ms  50-100 points
 243
 244            - **Output Queue**: O(k) space with automatic size management via maxlen
 245              Maximum size: (path_length_ms / output_nominal_period_ms) + 1
 246
 247            - **Cubic Spline Objects**: O(n) space for polynomial coefficients
 248              Three interpolators (lat, lon, alt) each storing n-segment splines
 249
 250            - **Numpy Arrays**: O(n) temporary space during spline creation
 251              Four arrays: times, latitudes, longitudes, altitudes
 252
 253            - **Geocode Object**: O(1) constant space for distance/bearing calculations
 254
 255            - **Local Variables**: O(1) constant space for iterators and temporaries
 256
 257        Memory Efficiency Features:
 258            - Automatic queue size limits prevent unbounded growth
 259            - Deque operations provide O(1) append/pop from both ends
 260            - Numpy arrays are reused and sized exactly to queue length
 261            - Old data automatically purged based on temporal windows
 262
 263        Steady-State Memory: In continuous operation, memory usage stabilizes
 264        at O(max_queue_sizes) regardless of total points processed over time.
 265
 266    Performance Optimizations:
 267        - **Deque Data Structure**: O(1) append/pop operations for queue management
 268        - **Numpy Vectorization**: Efficient array operations for coordinate processing
 269        - **Binary Search Splines**: O(log(n)) evaluation via scipy's optimized implementation
 270        - **Early Validation**: O(1) rejection of invalid points before expensive operations
 271        - **Bounded Queue Sizes**: Prevents memory growth in long-running applications
 272        - **Single-Pass Processing**: Minimizes data traversal and copying operations
 273
 274    Typical Use Cases:
 275        - GNSS trajectory smoothing for tracking applications
 276        - Temporal alignment of position data for synchronized processing
 277        - Generation of uniform sampling rates from irregular position updates
 278        - Motion parameter estimation from sparse position data
 279        - Real-time position interpolation for navigation systems
 280
 281    Thread Safety:
 282        This class is **not thread-safe**. External synchronization is required when
 283        accessing from multiple threads concurrently.
 284
 285    Examples:
 286        Basic usage for trajectory interpolation:
 287
 288        >>> config = PointInterpolationConfig(
 289        ...     path_length_ms=5000,
 290        ...     output_nominal_period_ms=100,
 291        ...     period_max_deviation_ms=20
 292        ... )
 293        >>> interpolator = PointInterpolation(config, logger)
 294        >>> result = interpolator.add_point(timestamp_ms, position_info)
 295        >>> if result == PointInterpolation.InsertPointResult.SUCCESS:
 296        ...     interpolated_positions = interpolator.get_interpolated_values()
 297
 298    Attributes:

 311        to 0.01 m/s units, and heading values are scaled to 0.1 degree units to match

 443    def _create_interpolators(self) -> tuple[CubicSpline, CubicSpline, CubicSpline]:
 444        """Create cubic spline interpolators for latitude, longitude, and altitude.
 445
 446        Processes the input points queue to extract time and position data, then
 447        creates three separate cubic spline interpolators for each spatial dimension.
 448        Uses pre-allocated arrays from the array cache for efficient data processing.
 449
 450        Returns:
 451            tuple[CubicSpline, CubicSpline, CubicSpline]: A tuple containing:
 452
 453                - Latitude interpolator (CubicSpline)
 454                - Longitude interpolator (CubicSpline)
 455                - Altitude interpolator (CubicSpline)
 456
 457        Note:
 458            If altitude is None for a position, it defaults to 0.
 459            Arrays are sliced to match the actual queue length to avoid using
 460            uninitialized cached array elements.
 461        """
 462        queue_len = len(self._input_points_queue)
 463        point_times_array = numpy.empty(queue_len, dtype=numpy.int64)
 464        latitudes_array = numpy.empty(queue_len, dtype=numpy.float64)
 465        longitudes_array = numpy.empty(queue_len, dtype=numpy.float64)
 466        altitudes_array = numpy.empty(queue_len, dtype=numpy.int32)
 467        # Single vectorized pass through queue to populate arrays directly
 468        for i, (time, position_info) in enumerate(self._input_points_queue):
 469            point_times_array[i] = time
 470            position = position_info.position
 471            latitudes_array[i] = position.latitude
 472            longitudes_array[i] = position.longitude
 473            altitudes_array[i] = position.altitude.value if position.altitude else 0
 474        return (
 475            CubicSpline(x=point_times_array, y=latitudes_array[:queue_len]),
 476            CubicSpline(x=point_times_array, y=longitudes_array[:queue_len]),
 477            CubicSpline(x=point_times_array, y=altitudes_array[:queue_len]),
 478        )

 480    def _modify_real_point(
 481        self,
 482        time_1: int,
 483        time_2: int,
 484        time_diff: int,
 485        previous_lat: float,
 486        previous_lon: float,
 487        position: PositionInfo,
 488    ) -> PositionInfo:
 489        """Modify a real point by calculating missing speed and heading information.
 490
 491        Takes an existing PositionInfo object and enriches it with computed speed and
 492        heading values if they are missing. If both speed and heading are already present,
 493        returns the position unchanged. The calculation is based on the distance and
 494        bearing between the previous position and the current position.
 495
 496        Time Complexity:
 497            O(1): All operations execute in constant time:
 498
 499                - Early return check: O(1)
 500                - Geocode setup: O(1) (tuple assignment)
 501                - Distance calculation: O(1) (haversine formula with trigonometric functions)
 502                - Bearing calculation: O(1) (atan2 and trigonometric functions)
 503                - Object creation: O(1)
 504                - Mathematical operations: O(1)
 505
 506        Space Complexity:
 507            O(1): Constant space usage:
 508
 509                - Fixed number of local variables (speed_obj, heading_obj)
 510                - Object creation uses constant memory regardless of input size
 511                - No dynamic data structures or recursion
 512                - Memory usage independent of position data complexity
 513
 514        Args:
 515            time_1 (int): Timestamp of the first position in milliseconds.
 516            time_2 (int): Timestamp of the second position in milliseconds.
 517            time_diff (int): Time difference in milliseconds between the previous and
 518                current positions. Used for speed calculation. Must be greater than 0
 519                for speed calculation, otherwise speed will remain None.
 520            previous_lat (float): Latitude coordinate of the previous position in decimal
 521                degrees. Used as reference point for distance and bearing calculations.
 522            previous_lon (float): Longitude coordinate of the previous position in decimal
 523                degrees. Used as reference point for distance and bearing calculations.
 524            position (PositionInfo): The current position information that may be missing
 525                speed and/or heading data. All other fields are preserved in the output.
 526
 527        Returns:
 528            PositionInfo: A new PositionInfo object with the same data as the input
 529            position, but with speed and heading filled in if they were missing:
 530
 531                - Speed: Calculated from distance divided by time difference, scaled
 532                  to 0.01 m/s units
 533                - Heading: Calculated as bearing from previous to current position,
 534                  scaled to 0.1 degree units
 535                - All other fields (stationInfo, objectId, positionType, time, position)
 536                  are copied unchanged from the input
 537
 538        Note:
 539            The method preserves the original position's positionType (e.g., real data)
 540            rather than marking it as estimated. Speed calculation uses the formula:
 541            distance * SPEED_SCALE_FACTOR / time_diff where SPEED_SCALE_FACTOR converts
 542            from m/ms to 0.01 m/s units. Heading is scaled by HEADING_SCALE_FACTOR to
 543            convert degrees to 0.1 degree units.
 544        """
 545        # Early return if both speed and heading are already present
 546        if position.speed and position.heading:
 547            return position
 548        self._geocode.point_a = (previous_lat, previous_lon)
 549        self._geocode.point_b = (
 550            position.position.latitude,
 551            position.position.longitude,
 552        )
 553        speed_obj = position.speed
 554        heading_obj = position.heading
 555        if (speed_obj is None) and (time_diff > 0):
 556            distance_m = self._geocode.get_distance(unit=DistanceUnit.METERS)
 557            value = int(distance_m * self._SPEED_SCALE_FACTOR) // time_diff
 558            try:
 559                speed_obj = Speed(
 560                    value=value,
 561                    confidence=None,
 562                )
 563            except ValidationError:
 564                self._logger.warning(
 565                    msg=f"Error calculating speed with true point {value}: "
 566                    f"{ItsTime.create_from_epoch_ms(epoch_ms=time_1)}, "
 567                    f"{ItsTime.create_from_epoch_ms(epoch_ms=time_2)}, "
 568                    f"distance: {distance_m} meters, "
 569                    f"latitude_a: {previous_lat}, "
 570                    f"longitude_a: {previous_lon}, "
 571                    f"latitude_b: {position.position.latitude}, "
 572                    f"longitude_b: {position.position.longitude}"
 573                )
 574                raise
 575        if heading_obj is None:
 576            bearing_deg = self._geocode.get_bearing(unit=AngleUnit.DEGREES)
 577            value = int(bearing_deg * self._HEADING_SCALE_FACTOR)
 578            try:
 579                heading_obj = Heading(value=value, confidence=None)
 580            except ValidationError:
 581                self._logger.warning(
 582                    msg=f"Error calculating heading with true point {value}: "
 583                    f"{ItsTime.create_from_epoch_ms(epoch_ms=time_1)}, "
 584                    f"{ItsTime.create_from_epoch_ms(epoch_ms=time_2)}, "
 585                    f"bearing: {bearing_deg} degrees, "
 586                    f"latitude_a: {previous_lat}, "
 587                    f"longitude_a: {previous_lon}, "
 588                    f"latitude_b: {position.position.latitude}, "
 589                    f"longitude_b: {position.position.longitude}"
 590                )
 591                raise
 592        return PositionInfo(
 593            stationInfo=position.stationInfo,
 594            objectId=position.objectId,
 595            positionType=position.positionType,
 596            time=position.time,
 597            position=position.position,
 598            speed=speed_obj,
 599            heading=heading_obj,
 600            driveDirection=None,
 601        )

 603    def _create_interpolated_position(
 604        self,
 605        station_info: Identification,
 606        object_id: Optional[int],
 607        time: int,
 608        current_lat: float,
 609        current_lon: float,

 612        previous_lat: float,

 615        """Create an interpolated position with calculated speed and heading.
 616
 617        Generates a new PositionInfo object for an interpolated position by calculating

 621
 622        Time Complexity:
 623            O(1): All operations are constant time:
 624
 625                - Geocode distance calculation: O(1) (haversine formula)
 626                - Geocode bearing calculation: O(1) (atan2 trigonometric function)
 627                - Object instantiation: O(1)
 628                - Mathematical operations: O(1)
 629
 630        Space Complexity:
 631            O(1): Constant space usage:
 632
 633                - Fixed number of local variables regardless of input size
 634                - Object creation uses constant memory
 635                - No data structures that grow with input parameters
 636
 637        Args:
 638            station_info (Identification): Station identification information for the position.
 639            object_id (Optional[int]): Optional object identifier for tracking purposes.
 640            time (int): Timestamp in milliseconds for the interpolated position.
 641            current_lat (float): Latitude coordinate of the current interpolated position.
 642            current_lon (float): Longitude coordinate of the current interpolated position.
 643            current_alt (int): Altitude value of the current interpolated position.
 644            time_diff (int): Time difference in milliseconds between previous and current positions.
 645                Used for speed calculation. If zero or negative, speed will be None.

 647                calculation.
 648            previous_lon (float): Longitude coordinate of the previous position for motion
 649                calculation.
 650
 651        Returns:
 652            PositionInfo: A new PositionInfo object containing:
 653
 654                - Station and object identification
 655                - Position type marked as POSITION_ESTIMATED

 658                - Calculated heading based on bearing between positions
 659                - Drive direction set to None
 660
 661        Note:
 662            Speed calculation uses the formula: distance * SPEED_SCALE_FACTOR / time_diff
 663            where SPEED_SCALE_FACTOR converts from m/ms to 0.01 m/s units.
 664            Heading is scaled by HEADING_SCALE_FACTOR to convert degrees to 0.1 degree units.
 665        """
 666        self._geocode.point_a = (previous_lat, previous_lon)
 667        self._geocode.point_b = (current_lat, current_lon)
 668        distance_m = self._geocode.get_distance(unit=DistanceUnit.METERS)
 669        bearing_deg = self._geocode.get_bearing(unit=AngleUnit.DEGREES)
 670        speed_obj: Optional[Speed] = None
 671        if time_diff > 0:
 672            value = int(distance_m * self._SPEED_SCALE_FACTOR) // time_diff
 673            try:
 674                speed_obj = Speed(
 675                    value=value,
 676                    confidence=None,
 677                )
 678            except ValidationError:
 679                self._logger.warning(
 680                    msg=f"Error calculating speed with interpolated point {value}: "
 681                    f"{ItsTime.create_from_epoch_ms(epoch_ms=time)}, "
 682                    f"diff: {time_diff} ms, "
 683                    f"distance: {distance_m} meters, "
 684                    f"latitude_a: {previous_lat}, "
 685                    f"longitude_a: {previous_lon}, "
 686                    f"latitude_b: {current_lat}, "
 687                    f"longitude_b: {current_lon}"
 688                )
 689                raise
 690        value = int(bearing_deg * self._HEADING_SCALE_FACTOR)
 691        try:
 692            heading_obj = Heading(value=value, confidence=None)
 693        except ValidationError:
 694            self._logger.warning(
 695                msg=f"Error calculating heading with interpolated point {value}: "
 696                f"{ItsTime.create_from_epoch_ms(epoch_ms=time)}, "
 697                f"diff: {time_diff} ms, "
 698                f"bearing: {bearing_deg} degrees, "
 699                f"latitude_a: {previous_lat}, "
 700                f"longitude_a: {previous_lon}, "
 701                f"latitude_b: {current_lat}, "
 702                f"longitude_b: {current_lon}"
 703            )
 704            raise
 705        return PositionInfo(
 706            stationInfo=station_info,
 707            objectId=object_id,
 708            positionType=PositionType.POSITION_ESTIMATED,

 710            position=ReferencePosition(
 711                latitude=current_lat,
 712                longitude=current_lon,
 713                positionConfidence=None,
 714                altitude=Altitude(value=current_alt, confidence=None),
 715            ),
 716            speed=speed_obj,
 717            heading=heading_obj,
 718            driveDirection=None,
 719        )

 721    def _end_time_calculation(self, start_time: int, new_point_time: int) -> int:
 722        """Calculate the end time for interpolation based on start time and new point
 723        time.
 724
 725        This method determines the appropriate end time by considering the output nominal period
 726        and minimum period constraints. It ensures the end time aligns with period boundaries
 727        while respecting minimum period requirements.
 728
 729        Args:
 730            start_time (int): The starting time point in milliseconds
 731            new_point_time (int): The new data point time in milliseconds
 732
 733        Returns:
 734            int: The calculated end time in milliseconds, adjusted to period boundaries
 735
 736        Notes:
 737            - The calculation considers self._output_nominal_period_ms for period alignment
 738            - If the remainder period is >= self._min_period_ms, it's included in the result
 739            - The result is always >= start_time and <= new_point_time
 740        """
 741        period = self._output_nominal_period_ms
 742        range_long = new_point_time - start_time
 743        range_step_multiple = (range_long // period) * period
 744        range_rest = range_long % period
 745        if range_rest >= self._min_period_ms:
 746            range_step_multiple += range_rest
 747        return start_time + range_step_multiple

 749    def _update_output_queue(self) -> None:
 750        """Update the output queue with interpolated and real position data.
 751
 752        This method is the core of the interpolation system. It processes the input points
 753        queue to generate a time-aligned sequence of position data in the output queue.
 754        The method creates cubic spline interpolators for spatial coordinates and then
 755        generates positions at regular time intervals, using either real GNSS points
 756        (when available within tolerance) or interpolated positions.
 757
 758        The algorithm works in several phases:
 759
 760            #. Creates cubic spline interpolators for latitude, longitude, and altitude
 761            #. Determines the time range for output generation based on existing output
 762            #. Iterates through the time range at nominal period intervals
 763            #. For each time point, decides whether to use a real GNSS point or interpolate
 764            #. Calculates missing speed/heading data and adds the position to output queue
 765
 766        Time Complexity:
 767            O(n + k*log(n) + k*m): Where n = input queue size, k = output points generated,
 768            m = average input points processed per output point:
 769
 770                - Interpolator creation: O(n) for cubic spline setup
 771                - Time range iteration: O(k) where k = (end_time - start_time) / nominal_period
 772                - Spline evaluation per point: O(log(n)) due to binary search in spline
 773                - Input queue traversal: O(k*m) in worst case, but typically O(n) total
 774                - Position object creation: O(k) for k output points
 775                - Queue operations: O(k) for deque appendleft operations
 776
 777            Typical case: O(n + k*log(n)) where k is proportional to the time span
 778            Worst case: O(n + k*log(n) + n*k) if input points are densely packed
 779
 780        Space Complexity:
 781            O(n + k): Where n = input queue size, k = output points generated:
 782
 783                - Interpolator storage: O(n) for cubic spline coefficients
 784                - Local variables: O(1) constant space
 785                - Output queue growth: O(k) for new position objects
 786                - Iterator and temporary objects: O(1) constant space
 787
 788            The output queue has a maximum size limit, so in steady state the space
 789            complexity becomes O(n + maxlen) where maxlen is the queue size limit.
 790
 791        Returns:
 792            None
 793
 794        Note:
 795            The method uses a tolerance-based approach to decide between real and
 796            interpolated points. Real points are preferred when their timestamp
 797            is within _period_max_deviation_ms of the target time. The output queue
 798            uses appendleft to maintain reverse chronological order (newest first).
 799
 800            The cubic spline interpolation provides smooth position transitions but
 801            requires at least 4 points for proper operation. Missing speed and heading
 802            data is calculated based on distance and bearing between consecutive positions.
 803        """
 804        (lat_interpolator, lon_interpolator, alt_interpolator) = (
 805            self._create_interpolators()
 806        )
 807        input_queue_iterator = iter(self._input_points_queue)
 808        input_time, input_position = next(input_queue_iterator)
 809        last_time: Optional[int] = None
 810        last_lat: Optional[float] = None
 811        last_lon: Optional[float] = None
 812        # Start and end range output queue
 813        if len(self._output_points_queue) == 0:
 814            # Start from oldest input point
 815            start_time = input_time
 816        else:
 817            # Start after the last output point
 818            tmp_time, tmp_position = self._output_points_queue[0]
 819            start_time = tmp_time + self._output_nominal_period_ms
 820            last_time = tmp_time
 821            last_lat = tmp_position.position.latitude
 822            last_lon = tmp_position.position.longitude
 823        end_time = self._end_time_calculation(
 824            start_time=start_time, new_point_time=self._input_points_queue[-1][0]
 825        )
 826        time_range = range(
 827            start_time,
 828            end_time + self._output_nominal_period_ms,
 829            self._output_nominal_period_ms,
 830        )
 831        for time in time_range:
 832            new_position: Optional[PositionInfo] = None
 833            while True:
 834                if abs(time - input_time) <= self._period_max_deviation_ms:
 835                    if last_time and last_lat and last_lon:
 836                        new_position = self._modify_real_point(
 837                            time_1=time,
 838                            time_2=input_time,
 839                            time_diff=abs(input_time - last_time),
 840                            previous_lat=last_lat,
 841                            previous_lon=last_lon,
 842                            position=input_position,
 843                        )
 844                    else:
 845                        new_position = input_position
 846                    try:
 847                        input_time, input_position = next(input_queue_iterator)
 848                    except StopIteration:
 849                        pass
 850                    break
 851                if input_time > time:
 852                    break
 853                input_time, input_position = next(input_queue_iterator)
 854            if new_position is None:
 855                # Create interpolated position
 856                _, last_input_position = self._input_points_queue[-1]
 857                current_lat = float(lat_interpolator(x=time))
 858                current_lon = float(lon_interpolator(x=time))
 859                new_position = self._create_interpolated_position(
 860                    station_info=last_input_position.stationInfo,
 861                    object_id=last_input_position.objectId,
 862                    time=time,
 863                    current_lat=current_lat,
 864                    current_lon=current_lon,
 865                    current_alt=int(alt_interpolator(x=time)),
 866                    time_diff=(abs(last_time - time) if last_time else 0),
 867                    previous_lat=last_lat if last_lat else current_lat,
 868                    previous_lon=last_lon if last_lon else current_lon,
 869                )
 870            last_time = ItsTime.create_from_iso_time(
 871                time_str=new_position.time
 872            ).epoch_ms
 873            last_lat = new_position.position.latitude
 874            last_lon = new_position.position.longitude
 875            self._output_points_queue.appendleft((last_time, new_position))

 877    def _check_position_coherence(
 878        self,
 879        current_position_time: int,
 880        current_position: PositionInfo,
 881        last_position_time: int,
 882        last_position: PositionInfo,
 883    ) -> "PointInterpolation.InsertPointResult":
 884        """Checks the coherence of a new position based on its timestamp and spatial
 885        relation to the last recorded position.
 886
 887        This method performs the following checks:
 888            - Ensures the time difference between the current and last position is not
 889              too short (below minimum period).
 890            - Ensures the time difference is not too long (above maximum period).
 891            - Calculates the speed between the two positions and ensures it does not
 892              exceed a defined threshold.
 893
 894        If any check fails, an appropriate result is returned and a warning is logged.
 895
 896        Args:
 897            current_position_time (int): Epoch time in milliseconds of the current position.
 898            current_position (PositionInfo): Information about the current position.
 899            last_position_time (int): Epoch time in milliseconds of the last recorded position.
 900            last_position (PositionInfo): Information about the last recorded position.
 901
 902        Returns:
 903            PointInterpolation.InsertPointResult: Result indicating the outcome of
 904            the coherence check.
 905        """
 906        diff_time = current_position_time - last_position_time
 907        if diff_time < self._min_period_ms:
 908            self._logger.warning(
 909                msg=f"Position time {ItsTime.create_from_epoch_ms(epoch_ms=current_position_time)}"
 910                "is too close to or older than "
 911                f"the last recorded time "
 912                f"{ItsTime.create_from_epoch_ms(epoch_ms=last_position_time)}, ignoring point"
 913            )
 914            return PointInterpolation.InsertPointResult.PERIOD_TOO_SHORT
 915        if diff_time >= self._max_period_ms:
 916            self._logger.warning(
 917                msg=f"Position time {ItsTime.create_from_epoch_ms(epoch_ms=current_position_time)} "
 918                f"is too far (max {self._max_period_ms} ms) from the last recorded time "
 919                f"{ItsTime.create_from_epoch_ms(epoch_ms=last_position_time)}, clearing history"
 920            )
 921            return PointInterpolation.InsertPointResult.PERIOD_TOO_LONG
 922        self._geocode.point_a = (
 923            last_position.position.latitude,
 924            last_position.position.longitude,
 925        )
 926        self._geocode.point_b = (
 927            current_position.position.latitude,
 928            current_position.position.longitude,
 929        )
 930        # Speed calculation in m/s
 931        speed = (self._geocode.get_distance(DistanceUnit.METERS) * 1000) / diff_time
 932        if speed > self._SPEED_COHERENCE_THRESHOLD:
 933            self._logger.warning(
 934                msg=f"Position speed {speed} m/s exceeds threshold "
 935                f"{self._SPEED_COHERENCE_THRESHOLD} m/s, ignoring point"
 936            )
 937            return PointInterpolation.InsertPointResult.POSITION_ERROR
 938        return PointInterpolation.InsertPointResult.SUCCESS

 940    def add_point(
 941        self, position_time: int, position: PositionInfo
 942    ) -> "PointInterpolation.InsertPointResult":
 943        """Add a new position point to the interpolation system and trigger output
 944        generation.
 945
 946        This method is the primary interface for feeding position data into the interpolation
 947        system. It validates the temporal consistency of incoming points, manages the input
 948        queue by removing outdated points, and triggers the interpolation process when
 949        sufficient data is available. The method implements several validation checks to
 950        ensure data quality and temporal coherence.
 951
 952        The algorithm performs the following operations:
 953
 954            #. **Empty Queue Handling**: If the input queue is empty, adds the point and
 955               returns FILLING_PERIOD status.
 956            #. **Temporal Validation**: Checks if the new point's timestamp is within
 957               acceptable bounds relative to the last point.
 958            #. **Gap Detection**: Identifies data gaps that are too large and clears
 959               history when necessary.
 960            #. **Queue Management**: Removes outdated points that fall outside the
 961               minimum path length window.
 962            #. **Interpolation Trigger**: Calls _update_output_queue() when sufficient
 963               data is available for interpolation.
 964
 965        Time Complexity:
 966            The time complexity varies based on the validation outcome:
 967
 968            - **Best Case - Early Return**: O(1) when validation fails (PERIOD_TOO_SHORT
 969              or empty queue scenarios)
 970            - **Gap Detection Case**: O(1) for clearing history and adding point
 971              (PERIOD_TOO_LONG)
 972            - **Successful Case**: O(n + k*log(n) + m) where:
 973
 974                - n = current input queue size
 975                - k = number of output points generated by _update_output_queue()
 976                - m = number of outdated points removed from queue (typically small)
 977                  so we can consider as O(1).
 978
 979              The dominant factors are:
 980
 981                - Queue cleanup: O(m) for removing outdated points. Typical we can
 982                  consider as O(1).
 983                - _update_output_queue(): O(n + k*log(n)) for interpolation process
 984                - Deque operations: O(1) per append/popleft operation
 985
 986            **Typical Case**: O(n + k*log(n)) where k is proportional to the time span
 987            covered by new interpolated points.
 988
 989        Space Complexity:
 990            O(n + k) where:
 991
 992            - n = input queue size (bounded by path length configuration)
 993            - k = output points generated and stored in output queue
 994
 995            **Memory Usage Details**:
 996
 997                - Input queue: O(n) for storing (timestamp, PositionInfo) tuples
 998                - Output queue: O(k) for interpolated positions (with maxlen limit)
 999                - Interpolator objects: O(n) for cubic spline coefficients
1000                - Local variables: O(1) constant space
1001                - Temporary objects: O(1) for validation and calculations
1002
1003            In steady state, both input and output queues have maximum size limits,
1004            making the practical space complexity O(max_input + max_output).
1005
1006        Args:
1007            position_time (int): Timestamp of the position in milliseconds since epoch.
1008                Must be greater than the last point's timestamp by at least _min_period_ms
1009                and less than _max_period_ms to maintain temporal consistency.
1010            position (PositionInfo): Complete position information including coordinates,
1011                speed, heading, and metadata. The position will be validated and may
1012                trigger interpolation calculations.
1013
1014        Returns:
1015            PointInterpolation.InsertPointResult: Enumeration indicating the result:
1016
1017                - **SUCCESS**: Point added successfully and interpolation was performed.
1018                  Output queue contains new interpolated positions.
1019                - **FILLING_PERIOD**: Point added but insufficient data for interpolation.
1020                  More points needed to reach minimum path length requirement.
1021                - **PERIOD_TOO_SHORT**: Point rejected due to timestamp being too close
1022                  to or older than the previous point. No changes made to queues.
1023                - **PERIOD_TOO_LONG**: Large temporal gap detected. History cleared
1024                  and point added as new starting point for future interpolation.
1025                - **POSITION_ERROR**: Point rejected due to excessive speed or other
1026                  factors.
1027
1028        Note:
1029            The method maintains temporal ordering and data quality by rejecting points
1030            that would compromise the interpolation accuracy. When PERIOD_TOO_LONG is
1031            returned, the entire history is cleared to prevent interpolation across
1032            large data gaps that could produce unrealistic results.
1033
1034            The interpolation process is triggered only when there are sufficient points
1035            spanning the minimum path length, ensuring the cubic spline interpolators
1036            have adequate data for smooth curve generation.
1037        """
1038        if len(self._input_points_queue) == 0:
1039            self._input_points_queue.append((position_time, position))
1040            self._logger.debug(
1041                msg=f"First position: time {ItsTime.create_from_epoch_ms(epoch_ms=position_time)}"
1042            )
1043            return PointInterpolation.InsertPointResult.FILLING_PERIOD
1044        last_position_time, last_position = self._input_points_queue[-1]
1045        match self._check_position_coherence(
1046            current_position_time=position_time,
1047            current_position=position,
1048            last_position_time=last_position_time,
1049            last_position=last_position,
1050        ):
1051            case (
1052                PointInterpolation.InsertPointResult.SUCCESS
1053                | PointInterpolation.InsertPointResult.FILLING_PERIOD
1054            ):
1055                self._error_consecutive_count = 0
1056            case PointInterpolation.InsertPointResult.PERIOD_TOO_SHORT:
1057                return PointInterpolation.InsertPointResult.PERIOD_TOO_SHORT
1058            case PointInterpolation.InsertPointResult.PERIOD_TOO_LONG:
1059                self._clear_history()
1060                self._error_consecutive_count = 0
1061                self._input_points_queue.append((position_time, position))
1062                return PointInterpolation.InsertPointResult.PERIOD_TOO_LONG
1063            case PointInterpolation.InsertPointResult.POSITION_ERROR | _:
1064                if (
1065                    self._error_consecutive_count
1066                    < PointInterpolation._ERROR_CONSECUTIVE_THRESHOLD
1067                ):
1068                    self._error_consecutive_count += 1
1069                else:
1070                    self._logger.warning(
1071                        msg="Position error threshold exceeded: "
1072                        f"{self._error_consecutive_count} errors"
1073                    )
1074                    self._clear_history()
1075                    self._error_consecutive_count = 0
1076                    self._input_points_queue.append((position_time, position))
1077                return PointInterpolation.InsertPointResult.POSITION_ERROR
1078        self._input_points_queue.append((position_time, position))
1079        cutoff_time = position_time - self._min_path_length_ms
1080        (first_time, _) = self._input_points_queue[0]
1081        if first_time > cutoff_time:
1082            self._logger.debug(
1083                msg="The queue isn't full: "
1084                f"last time {ItsTime.create_from_epoch_ms(epoch_ms=position_time)}, "
1085                f"first time {ItsTime.create_from_epoch_ms(epoch_ms=first_time)}"
1086            )
1087            return PointInterpolation.InsertPointResult.FILLING_PERIOD
1088        # Removed too old points
1089        while (
1090            len(self._input_points_queue) > 1
1091            and self._input_points_queue[1][0] <= cutoff_time
1092        ):
1093            (time, _) = self._input_points_queue.popleft()
1094            self._logger.debug(
1095                msg=f"Removed too older point {ItsTime.create_from_epoch_ms(epoch_ms=time)} "
1096                f"than {ItsTime.create_from_epoch_ms(epoch_ms=position_time)}, "
1097                f"new length {len(self._input_points_queue)}"
1098            )
1099        self._update_output_queue()
1100        return PointInterpolation.InsertPointResult.SUCCESS
