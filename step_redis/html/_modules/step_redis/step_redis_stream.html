  1"""Redis Stream Module for asynchronous message processing.
  2
  3This module provides a comprehensive interface for working with Redis Streams using
  4asynchronous context management. Redis Streams are an append-only data structure
  5that allows for high-throughput pub/sub messaging with consumer groups and
  6persistence capabilities.
  7
  8The StepRedisStream class offers functionality for:
  9
 10    - Creating and managing Redis streams with consumer groups
 11    - Publishing messages to streams with configurable max length
 12    - Consuming messages from streams with proper acknowledgment handling
 13    - Setting and managing Time-To-Live (TTL) for streams
 14    - Connection health monitoring via ping commands
 15
 16This module is designed to be used in asynchronous Python applications that require
 17reliable message passing with persistence and consumer group capabilities. It supports
 18the async/await syntax and is intended to be used as an async context manager.
 19
 20Example:
 21    >>> import asyncio
 22    ... import logging
 23    ... from step_redis.step_redis_pool import StepRedisPool
 24    ... from step_redis.step_redis_stream import StepRedisStream
 25    ...
 26    ... async def main():
 27    ...    #Setup logging
 28    ...    logging.basicConfig(level=logging.DEBUG)
 29    ...    logger = logging.getLogger("redis_example")
 30    ...
 31    ...    # Create a Redis pool
 32    ...    pool = StepRedisPool(host="localhost", port=6379, db=0)
 33    ...
 34    ...    # Use the stream as a context manager
 35    ...    async with StepRedisStream(pool, logger) as stream:
 36    ...        # Create a stream with a consumer group
 37    ...        await stream.create_input_stream("my_stream", "my_group", ttl_seconds=3600)
 38    ...
 39    ...        # Publish a message
 40    ...        await stream.publish_message("my_stream", "Hello, Redis!", 1000)
 41    ...
 42    ...        # Consume a message
 43    ...        message = await stream.receive_message(
 44    ...            "my_stream", "my_group", "consumer1", 5000, ttl_seconds=3600
 45    ...        )
 46    ...        print(f"Received: {message}")
 47    ...
 48    ... asyncio.run(main())
 49"""

 64class StepRedisStream:
 65    """A class for interacting with Redis streams using async context management. This
 66    class provides methods for creating Redis streams, publishing messages to them, and
 67    consuming messages from them. It's designed to be used as an async context manager.
 68
 69    Example:
 70        >>> async with StepRedisStream(redis_pool, logger) as stream:
 71        ...    await stream.create_input_stream("my_stream", "my_group")
 72        ...    await stream.publish_message("my_stream", "Hello, Redis!", 1000)
 73        ...    message = await stream.receive_message("my_stream", "my_group", "consumer1", 5000)
 74    Attributes:
 75        logger (Logger): Logger instance for recording events and debug information.
 76        redis (Redis): Redis client instance used for stream operations.

101    async def __aexit__(self, exc_type: Any, exc_value: Any, traceback: Any) -> None:
102        """Asynchronously exits the Redis stream context manager, closing the Redis
103        connection.
104
105        This method is called when exiting the context manager using an 'async with' statement.
106
107        Args:
108            exc_type (Any): The exception type if an exception was raised in the context,

125    async def create_input_stream(
126        self, stream_id: str, group_id: str, ttl_seconds: Optional[int] = None
127    ) -> None:
128        """Creates an input stream with a consumer group in Redis.
129
130        This method creates a new Redis stream with a specified ID and associates it with a
131        consumer group. If the consumer group already exists, the method logs this information
132        and continues without error. Optionally, a time-to-live (TTL) can be set for the stream.
133
134        Args:
135            stream_id: The identifier for the Redis stream to create
136            group_id: The identifier for the consumer group to create/associate with the stream
137            ttl_seconds: Optional; The number of seconds until the stream expires, if specified
138
139        Returns:
140            None
141
142        Raises:
143            RedisResponseError: If there's an error with the Redis operation other than the consumer
144                group already existing
145        """
146        try:
147            response = await self._redis.xgroup_create(
148                name=stream_id,
149                groupname=group_id,
150                mkstream=True,
151            )
152            self._logger.debug(
153                msg=f"Created input stream {response}: {stream_id} with group {group_id}"
154            )
155        except RedisResponseError as e:
156            if "BUSYGROUP Consumer Group name already exists" in str(object=e):
157                self._logger.info(msg=f"Consumer group '{group_id}' already exists")
158            else:
159                raise
160        finally:
161            if ttl_seconds is not None:
162                await self.update_stream_ttl(
163                    stream_id=stream_id, ttl_seconds=ttl_seconds
164                )

166    async def update_stream_ttl(self, stream_id: str, ttl_seconds: int) -> bool:
167        """Update the Time-To-Live (TTL) expiration for an existing Redis stream.
168
169        This method updates the TTL expiration for an existing Redis stream. It uses the
170        Redis EXPIRE command which will either set a new TTL if none exists, or replace an
171        existing TTL with the new value. This is useful for refreshing the expiration time
172        for active streams that should persist longer.
173
174        Args:
175            stream_id (str): The identifier of the Redis stream to update TTL for
176            ttl_seconds (optional[int]): The number of seconds until the stream should expire
177                Must be a positive integer
178
179        Returns:
180            bool: True if the TTL was successfully updated, False otherwise
181            (False typically means the stream key doesn't exist)
182
183        Raises:
184            ValueError: If ttl_seconds is less than or equal to zero
185        """
186        if ttl_seconds <= 0:
187            raise ValueError("TTL seconds must be a positive integer")
188        self._logger.debug(
189            msg=f"Updating TTL to {ttl_seconds} seconds for stream {stream_id}"
190        )
191        # The Redis EXPIRE command returns 1 if the timeout was set, 0 if the key doesn't exist
192        result = int(await self._redis.expire(name=stream_id, time=ttl_seconds))
193        if result != 1:
194            self._logger.warning(
195                msg=f"Failed to update TTL for stream {stream_id}, unexpected error"
196            )
197            return False
198        self._logger.debug(msg=f"Successfully updated TTL for stream {stream_id}")
199        return True

201    async def get_stream_ttl(self, stream_id: str) -> int:
202        """Get the remaining Time-To-Live (TTL) in seconds for a Redis stream.
203
204        This method retrieves the remaining TTL for a Redis stream. It uses the Redis TTL
205        command which returns the remaining time to live for a key in seconds.
206
207        Args:
208            stream_id (str): The identifier of the Redis stream to check
209
210        Returns:
211            int: The number of seconds until the stream expires.
212            Returns -2 if the stream doesn't exist.
213            Returns -1 if the stream exists but has no associated TTL (will not expire).
214
215        Note:
216            A return value of -2 indicates the key does not exist.
217            A return value of -1 indicates the key exists but has no associated expiration.
218        """
219        self._logger.debug(msg=f"Getting TTL for stream {stream_id}")
220        ttl = int(await self._redis.ttl(name=stream_id))
221        if ttl == -2:
222            self._logger.debug(msg=f"Stream {stream_id} does not exist")
223        elif ttl == -1:
224            self._logger.debug(
225                msg=f"Stream {stream_id} exists but has no expiration set"
226            )
227        else:
228            self._logger.debug(msg=f"Stream {stream_id} will expire in {ttl} seconds")
229        return ttl

248    async def publish_message(
249        self, stream_id: str, message: str, stream_max_len: int
250    ) -> None:
251        """Publish a message to a Redis stream.
252
253        Args:
254            stream_id (str): The identifier of the Redis stream.
255            message (str): The message to publish.
256            stream_max_len (int): Truncate old stream members beyond this size.
257
258        Returns:
259            None
260
261        Note:
262            This method adds a message to a Redis stream with the given ID.
263            The message is added as a field named 'payload' in the stream entry.
264            The stream length is limited to stream_max_len entries.
265        """
266        self._logger.debug(msg=f"Publish message to Redis: {message}")
267        response = await self._redis.xadd(
268            name=stream_id,
269            fields={"payload": message},
270            maxlen=stream_max_len,
271            approximate=True,
272        )
273        self._logger.debug(msg=f"Published message {response} to {stream_id} stream")

275    async def receive_message(
276        self,
277        stream_id: str,
278        group_id: str,
279        consumer_name: str,
280        block_timeout: int,
281        ttl_seconds: Optional[int] = None,
282    ) -> str:
283        """Receives a message from a specified Redis stream using consumer groups.
284
285        This method continuously attempts to read a message from a Redis stream using the specified
286        consumer group and consumer.
287        It will keep trying until a message is successfully received and processed. If a TTL
288        is provided, it updates the stream's time-to-live after each attempt.
289
290        Args:
291            stream_id (str): The ID of the Redis stream to read from.
292            group_id (str): The consumer group ID.
293            consumer_name (str): The name of the consumer in the group.
294            block_timeout (int): Maximum time in milliseconds to block waiting for a message.
295            ttl_seconds (Optional[int], optional): Time-to-live in seconds to set for the stream.
296                If provided, the stream's TTL will be updated after each read attempt.
297                Defaults to None.
298
299        Returns:
300            str: The payload of the received message.
301
302        Raises:
303            RedisTimeoutError: Caught and logged when the block timeout is reached.
304            KeyError: Caught and logged when there's an issue with the response structure.
305
306        Note:
307            This method uses 'noack=True' which means that messages are automatically acknowledged
308            upon delivery and not added to the pending entries list.
309        """
310        message: str
311        while True:
312            try:
313                response = await self._redis.xreadgroup(
314                    groupname=group_id,
315                    consumername=consumer_name,
316                    streams={stream_id: ">"},
317                    block=block_timeout,
318                    count=1,
319                    noack=True,
320                )
321                self._logger.debug(msg=f"Received response: {response}")
322                if stream_id in response:
323                    message_id = str(object=response[stream_id][0][0][0])
324                    message = str(object=response[stream_id][0][0][1]["payload"])
325                    self._logger.debug(
326                        msg=f"Received message {message_id} from Redis: {message}"
327                    )
328                    break
329                self._logger.debug(msg=f"No stream {stream_id} in the response")
330            except RedisTimeoutError as e:
331                self._logger.debug(msg=f"{e}")
332            except KeyError as e:
333                self._logger.warning(msg=f"KeyError: {e}")
334            finally:
335                if ttl_seconds is not None:
336                    await self.update_stream_ttl(
337                        stream_id=stream_id, ttl_seconds=ttl_seconds
338                    )
339        return message
