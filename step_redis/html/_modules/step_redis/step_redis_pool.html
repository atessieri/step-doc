  1"""Redis connection pool management for asyncio applications.
  2
  3This module provides a specialized Redis connection pool implementation that extends the
  4BlockingConnectionPool from redis-py with enhanced features for asyncio applications. It
  5offers both SSL and non-SSL connections to Redis servers with configurable TCP keepalive
  6settings to maintain persistent connections, particularly useful for long-running services.
  7
  8The module consists of two main components:
  9
 10    #. StepRedisPoolParams: A dataclass that encapsulates all configuration parameters for
 11    the Redis connection pool, including network settings, authentication credentials,
 12    SSL/TLS configuration, and pool behavior settings.
 13
 14    #. StepRedisPool: A connection pool implementation that leverages the asyncio capabilities
 15    of redis-py and provides context management for proper resource cleanup through the
 16    async context manager protocol.
 17
 18Key Features:
 19
 20    - Comprehensive SSL/TLS support for secure Redis connections
 21    - TCP keepalive configuration for connection stability
 22    - Configurable connection limits and timeouts
 23    - Proper resource management through async context managers
 24    - Automatic response decoding
 25    - Integrated logging support
 26
 27Example:
 28    >>> import asyncio
 29    ... from logging import getLogger
 30    ... from step_redis.step_redis_pool import StepRedisPoolParams, StepRedisPool
 31    ...
 32    ... async def main():
 33    ...     # Configure Redis connection parameters
 34    ...     params = StepRedisPoolParams(
 35    ...         host="redis.example.com",
 36    ...         port=6379,
 37    ...         username="user",  # Optional
 38    ...         password="password",
 39    ...         db=0,
 40    ...         ssl_enable=True,
 41    ...         # ... other configuration parameters
 42    ...     )
 43    ...
 44    ...     logger = getLogger("redis_connection")
 45    ...
 46    ...     # Use the connection pool with async context manager
 47    ...     async with StepRedisPool(params=params, logger=logger) as pool:
 48    ...         # Create a Redis client using the pool
 49    ...         redis = Redis.from_pool(pool)
 50    ...         # Perform Redis operations
 51    ...         await redis.set("key", "value")
 52    ...         value = await redis.get("key")
 53    ...
 54    ... asyncio.run(main())
 55"""

118class StepRedisPool(BlockingConnectionPool):
119    """A custom Redis connection pool class that extends BlockingConnectionPool and
120    provides context management capabilities with async support. This class provides a
121    blocking connection pool for Redis connections with support for both SSL and non-SSL
122    connections. It configures connection parameters based on the provided
123    StepRedisPoolParams and includes TCP keepalive settings for maintaining persistent
124    connections. The class implements the asynchronous context manager protocol
125    (`__aenter__` and `__aexit__`), allowing it to be used with the async with statement
126    for proper resource management.
127
128    Attributes:
129        params (StepRedisPoolParams): Configuration parameters for the Redis connection pool,
130            including host, port, authentication details, SSL settings, timeouts, and other
131            connection parameters.
132        logger (Logger): Logger instance for recording debug and error information related to
133            the connection pool lifecycle.
134
135    Example:
136        >>> redis_params = StepRedisPoolParams(
137        ...    host="redis.example.com",
138        ...    port=6379,
139        ...    ssl_enable=True,
140        ...    # ... other parameters
141        ... logger = Logger("redis_connection")
142        ... async with StepRedisPool(params=redis_params, logger=logger) as redis_pool:
143        ...    # Use the redis pool here
144        ...    # Connections are automatically managed and cleaned up
145
146    Notes:
147        - When SSL is enabled, the pool uses SSLConnection and configures all required SSL
148          parameters
149        - When SSL is disabled, the pool uses standard Connection
150        - TCP keepalive settings are configured for both connection types
151        - All responses are automatically decoded (decode_responses=True)
152    """
153

222    async def __aexit__(self, exc_type: Any, exc: Any, tb: Any) -> None:
223        """Context manager exit method for asynchronous operation.
224
225        This method is called when exiting an asynchronous context manager (i.e., when exiting a
226        'async with' block). It ensures proper cleanup of the Redis connection pool.
227
228        Args:
229            exc_type (Any): The exception type if an exception was raised in the context, otherwise
230                None.
231            exc (Any): The exception instance if an exception was raised in the context, otherwise
232                None.
233            tb (Any): The traceback if an exception was raised in the context, otherwise None.
234
235        Returns:
236            None: Explicitly returns None to ensure that any exceptions raised in the context are
237            propagated.
238        """
239        self._logger.debug(msg="Redis pool context manager exited")
240        await self.aclose()
241        self._logger.debug(msg="Redis pool closed")
242        return None  # Explicitly return None to ensure exceptions propagate
