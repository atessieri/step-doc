  1"""Redis Integration Module for STEP Crash Prediction Framework.
  2
  3This module provides Redis-based integration components for the STEP framework,
  4enabling real-time message streaming and Crash Prediction data processing through
  5Redis Streams. It implements specialized Redis connection pooling and stream
  6management tailored for the STEP Crash Prediction architecture.
  7
  8The module contains two main classes:
  9
 10    #. **StepCrashPredictionRedisPool**: A Redis connection pool manager that extends the base
 11       ``StepRedisPool`` with STEP-specific configurations. It provides async context
 12       manager support and optimized connection handling for the Crash Prediction workflow.
 13
 14    #. **StepCrashPredictionRedisStream**: A Redis stream handler that manages both input
 15       message consumption and output message publishing for Producer workflows.
 16       It handles CloudEvents-compliant messaging and automatic stream lifecycle management.
 17
 18Key Features:
 19    - Async context manager support for proper resource management
 20    - CloudEvents-compliant message formatting and validation
 21    - Automatic stream TTL management based on configuration timeouts
 22    - Consumer group management for reliable message processing
 23    - Position array data publishing to Crash Prediction streams
 24    - Comprehensive logging and error handling
 25
 26Usage Example:
 27    >>> async with StepCrashPredictionRedisPool(config, logger) as pool:
 28    ...     async with StepCrashPredictionRedisStream(
 29                consumer_name,
 30                config,
 31                logger,
 32                pool
 33            ) as stream:
 34    ...         await stream.create_client_input_stream()
 35    ...         message = await stream.receive_crash_prediction_message()
 36    ...         await stream.publish_station_collision_array(position_data)
 37
 38Note:
 39    This module is designed for high-throughput, real-time processing scenarios
 40    and uses memory-optimized designs with ``__slots__`` for performance.
 41"""

141class StepCrashPredictionRedisStream(StepRedisStream):
142    """
143    This class extends StepRedisStream to provide specialized functionality for handling
144    STEP Crash Prediction messages in a Redis streams environment. It manages both input message
145    consumption and output message publishing for Crash Prediction workflows.
146    The Crash Prediction handles:
147
148        - Creating and managing Redis input streams with consumer groups
149        - Receiving and validating Crash Prediction cloud events messages
150        - Publishing position array data to Crash Prediction streams
151        - Automatic TTL management based on configuration timeouts
152
153    Attributes:
154        consumer_name (str): Name of the consumer for the Redis stream.
155        config (RedisStreams): Configuration object containing Redis stream settings.
156        logger (Logger): Logger instance for recording events and errors.
157        pool (StepRedisPool): Redis connection pool for managing connections.
158    """
159
160    __slots__ = ("_consumer_name", "_config", "_stream_ttl")
161
162    def __init__(

167        pool: StepRedisPool,
168    ) -> None:

170        self._config = config
171        # Convert stream_ttl from milliseconds to seconds

247    async def receive_crash_prediction_message(
248        self,
249    ) -> CrashPredictionCloudEventsJsonRoot:
250        """Receives and processes a Crash Prediction message from Redis stream.
251
252        This method retrieves a message from the configured Redis stream using the
253        specified stream ID, group ID, and consumer name. The received message is
254        then validated and parsed into a CrashPredictionCloudEventsJsonRoot object.
255
256        Returns:
257            CrashPredictionCloudEventsJsonRoot: A validated cloud events message object
258                parsed from the received JSON message.
259
260        Note:
261            The method logs the received message content for debugging purposes.
262            Messages are received with the configured TTL and blocking timeout
263            settings from the Crash Prediction configuration.
264        """
265        message_json = await self.receive_message(
266            stream_id=self._config.input.streamId,
267            group_id=self._config.input.groupId,
268            consumer_name=self._consumer_name,
269            block_timeout=self._config.input.waitTimeout,
270            ttl_seconds=self._stream_ttl,
271        )
272        self._logger.info(msg=f"Received message: {message_json}")
273        return CrashPredictionCloudEventsJsonRoot.model_validate_json(
274            json_data=message_json, strict=True
275        )
