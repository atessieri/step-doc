374    def to_asn(self, reference_position: ReferencePosition) -> dict[Any, Any]:
375        """Convert the PerceivedObject instance to an ASN.1 dictionary representation.
376
377        Args:
378            reference_position (ReferencePosition): The reference position to use for coordinate
379                transformation.
380
381        Returns:
382            dict[Any, Any]: An ASN.1 dictionary representation of the PerceivedObject instance.
383        """
384        cartesian_coordinates = CartesianPosition3dWithConfidence.from_orm(
385            reference=reference_position,

402            message["objectDimensionX"] = self.objectDimensionX.to_asn()
403        if self.objectPerceptionQuality is not None:
404            message["objectPerceptionQuality"] = self.objectPerceptionQuality
405        return message
