294    def from_asn(asn_dict: dict[Any, Any]) -> "CamDataV1":
295        """Deserializes an ASN.1 dictionary into a CamDataV1 object.
296
297        The method validates protocol version, message ID and high frequency container type
298        before creating the CamDataV1 object from the ASN.1 dictionary structure.
299
300        Args:
301            asn_dict (dict[Any, Any]): ASN.1 dictionary containing CAM (Cooperative Awareness
302                Message) data. Expected to have header, cam and yawRate fields.
303
304        Returns:
305            CamDataV1: A new instance of CamDataV1 populated with the data from the ASN.1
306                dictionary.
307
308        Raises:
309            ValueError: If protocol version is invalid, message ID is incorrect, or high frequency
310                container type is invalid.
311        """
312        header = ItsPduHeader.from_asn(asn_dict=asn_dict["header"])
313        if header.protocol_version != CamDataV1._ASN_PAYLOAD_VERSION:
314            raise ValueError(  # compatible with python 3.11
315                "Invalid protocol version: %s. Expected: %s"  # pylint:disable=consider-using-f-string
316                % (header.protocol_version, CamDataV1._ASN_PAYLOAD_VERSION),
317            )
318        if header.message_type != MessageId.CAM_PDU:
319            raise ValueError(  # compatible with python 3.11
320                "Invalid message ID: %s. Expected: %s"  # pylint:disable=consider-using-f-string
321                % (header.message_type, MessageId.DENM_PDU),
322            )
323        basic_container: dict[Any, Any] = asn_dict["cam"]["camParameters"][
324            "basicContainer"
325        ]
326        basic_vehicle_container_high_frequency: Optional[dict[Any, Any]] = (
327            asn_dict["cam"]["camParameters"]["highFrequencyContainer"][1]
328            if asn_dict["cam"]["camParameters"]["highFrequencyContainer"][0]
329            == "basicVehicleContainerHighFrequency"
330            else None
331        )
332        basic_vehicle_container_low_frequency: Optional[dict[Any, Any]] = (
333            asn_dict["cam"]["camParameters"]["lowFrequencyContainer"][1]
334            if ("lowFrequencyContainer" in asn_dict["cam"]["camParameters"])
335            and (
336                asn_dict["cam"]["camParameters"]["lowFrequencyContainer"][0]
337                == "basicVehicleContainerLowFrequency"
338            )
339            else None
340        )
341        heading: Optional[Heading]
342        speed: Optional[Speed]
343        drive_direction: Optional[DriveDirection]
344        vehicle_length: Optional[VehicleLength]
345        vehicle_width: Optional[int]
346        longitudinal_acceleration: Optional[AccelerationComponent]
347        curvature: Optional[Curvature]
348        curvature_calculation_mode: Optional[CurvatureCalculationMode]
349        yaw_rate: Optional[YawRate]
350        if basic_vehicle_container_high_frequency is None:
351            heading = None
352            speed = None
353            drive_direction = None
354            vehicle_length = None
355            vehicle_width = None
356            longitudinal_acceleration = None
357            curvature = None
358            curvature_calculation_mode = None
359            yaw_rate = None
360        else:
361            heading = Heading.from_asn(
362                asn_dict=basic_vehicle_container_high_frequency["heading"],
363                value_field_name="headingValue",
364                conf_field_name="headingConfidence",
365            )
366            speed = Speed.from_asn(
367                asn_dict=basic_vehicle_container_high_frequency["speed"]
368            )
369            drive_direction = (
370                DriveDirection(basic_vehicle_container_high_frequency["driveDirection"])
371                if basic_vehicle_container_high_frequency["driveDirection"]
372                != DriveDirection.UNAVAILABLE.value
373                else None
374            )
375            vehicle_length = VehicleLength.from_asn(
376                asn_dict=basic_vehicle_container_high_frequency["vehicleLength"]
377            )
378            vehicle_width = (
379                int(basic_vehicle_container_high_frequency["vehicleWidth"])
380                if int(basic_vehicle_container_high_frequency["vehicleWidth"])
381                != CamDataV1._VEHICLE_WIDTH_VALUE_INVALID
382                else None
383            )
384            longitudinal_acceleration = AccelerationComponent.from_asn(
385                asn_dict=basic_vehicle_container_high_frequency[
386                    "longitudinalAcceleration"
387                ]
388            )
389            curvature = Curvature.from_asn(
390                asn_dict=basic_vehicle_container_high_frequency["curvature"]
391            )
392            curvature_calculation_mode = (
393                CurvatureCalculationMode(
394                    basic_vehicle_container_high_frequency["curvatureCalculationMode"]
395                )
396                if basic_vehicle_container_high_frequency["curvatureCalculationMode"]
397                != CurvatureCalculationMode.UNAVAILABLE.value
398                else None
399            )
400            yaw_rate = YawRate.from_asn(
401                asn_dict=basic_vehicle_container_high_frequency["yawRate"]
402            )
403        return CamDataV1(
404            creationTime=str(
405                object=ItsTime.create_from_delta_time(
406                    delta_time_ms=int(asn_dict["cam"]["generationDeltaTime"]),
407                    epoch=None,
408                    module=CamDataV1.DELTA_TIME_MS_MODULE,
409                )
410            ),
411            stationInfo=Identification(
412                stationId=header.station_id,
413                stationType=TrafficParticipantType.from_int(
414                    value=basic_container["stationType"],
415                ),
416            ),
417            position=ReferencePosition.from_asn(
418                asn_dict=basic_container["referencePosition"],
419                conf_semi_major_field_name="semiMajorAxisLength",
420                conf_semi_minor_field_name="semiMinorAxisLength",
421                conf_orient_field_name="semiMajorAxisOrientation",
422            ),
423            heading=heading,
424            speed=speed,
425            driveDirection=drive_direction,
426            vehicleLength=vehicle_length,
427            vehicleWidth=vehicle_width,
428            longitudinalAcceleration=longitudinal_acceleration,
429            curvature=curvature,
430            curvatureCalculationMode=curvature_calculation_mode,
431            yawRate=yaw_rate,
432            vehicleRole=(
433                VehicleRole(basic_vehicle_container_low_frequency["vehicleRole"])
434                if basic_vehicle_container_low_frequency is not None
435                else None
436            ),
437        )

439    def to_asn(self) -> dict[Any, Any]:
440        """Convert class instance to ASN.1 format dictionary.
441
442        Converts a CamDataV1 instance into a dictionary structure following the ASN.1
443        CAM specification format.
444        The message contains:
445
446            - header: Protocol information including version, message ID and station ID
447            - cam: The main CAM message content including:
448
449                - basicContainer: Basic station info and position
450                - highFrequencyContainer: Frequently updated vehicle data
451                - lowFrequencyContainer (optional): Less frequently updated vehicle data
452
453        Returns:
454            dict[Any, Any]: Dictionary representation of the CAM message in ASN.1 format
455        """
456        message: dict[Any, Any] = {
457            "header": ItsPduHeader(
458                protocol_version=CamDataV1._ASN_PAYLOAD_VERSION,
459                message_type=MessageId.CAM_PDU,
460                station_id=self.stationInfo.stationId,
461            ).to_asn(),
462            "cam": {
463                "generationDeltaTime": ItsTime.create_from_iso_time(
464                    time_str=self.creationTime,
465                ).get_delta_time_ms(
466                    module=CamDataV1.DELTA_TIME_MS_MODULE,
467                ),
468                "camParameters": {
469                    "basicContainer": {
470                        "stationType": TrafficParticipantType.to_int(
471                            participant_type=self.stationInfo.stationType
472                        ),
473                        "referencePosition": (
474                            self.position.to_asn(
475                                conf_semi_major_field_name="semiMajorAxisLength",
476                                conf_semi_minor_field_name="semiMinorAxisLength",
477                                conf_orient_field_name="semiMajorAxisOrientation",
478                            )
479                            if self.position is not None
480                            else ReferencePosition.to_asn_none(
481                                conf_semi_major_field_name="semiMajorAxisLength",
482                                conf_semi_minor_field_name="semiMinorAxisLength",
483                                conf_orient_field_name="semiMajorAxisOrientation",
484                            )
485                        ),
486                    },
487                    "highFrequencyContainer": (
488                        "basicVehicleContainerHighFrequency",
489                        {
490                            "heading": (
491                                self.heading.to_asn(
492                                    value_field_name="headingValue",
493                                    conf_field_name="headingConfidence",
494                                )
495                                if self.heading is not None
496                                else Heading.to_asn_none(
497                                    value_field_name="headingValue",
498                                    conf_field_name="headingConfidence",
499                                )
500                            ),
501                            "speed": (
502                                self.speed.to_asn()
503                                if self.speed is not None
504                                else Speed.to_asn_none()
505                            ),
506                            "driveDirection": (
507                                self.driveDirection.value
508                                if self.driveDirection is not None
509                                else DriveDirection.UNAVAILABLE.value
510                            ),
511                            "vehicleLength": (
512                                self.vehicleLength.to_asn()
513                                if self.vehicleLength is not None
514                                else VehicleLength.to_asn_none()
515                            ),
516                            "vehicleWidth": (
517                                self.vehicleWidth
518                                if self.vehicleWidth is not None
519                                else CamDataV1._VEHICLE_WIDTH_VALUE_INVALID
520                            ),
521                            "longitudinalAcceleration": (
522                                self.longitudinalAcceleration.to_asn()
523                                if self.longitudinalAcceleration is not None
524                                else AccelerationComponent.to_asn_none()
525                            ),
526                            "curvature": (
527                                self.curvature.to_asn()
528                                if self.curvature is not None
529                                else Curvature.to_asn_none()
530                            ),
531                            "curvatureCalculationMode": (
532                                self.curvatureCalculationMode.value
533                                if self.curvatureCalculationMode is not None
534                                else CurvatureCalculationMode.UNAVAILABLE.value
535                            ),
536                            "yawRate": (
537                                self.yawRate.to_asn()
538                                if self.yawRate is not None
539                                else YawRate.to_ans_none()
540                            ),
541                        },
542                    ),
543                },
544            },
545        }
546        if self.vehicleRole is not None:
547            message["cam"]["camParameters"]["lowFrequencyContainer"] = (
548                "basicVehicleContainerLowFrequency",
549                {
550                    "vehicleRole": self.vehicleRole.value,
551                    "exteriorLights": (b"\x00", 8),
552                    "pathHistory": [],
553                },
554            )
555        return message
